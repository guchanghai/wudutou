/**
 * It parses the url path and ojects selected from response, if valid.
 * @param result - Result from service to parse.
 * @returns Parsed response from the service.
 */
export const parseResponse = (
  result: string,
): { url: string; objectsSelected?: number } => {
  const pathStringify = result.match(/\{"path":(.)*}/);
  if (pathStringify) {
    try {
      const pathJSON = JSON.parse(pathStringify[0]);
      return {
        url: pathJSON.path,
        objectsSelected: pathJSON.discreteObjectsInInputMask,
      };
    } catch (e) {
      console.error("Error in parsing response");
    }
  }
  return { url: DEFAULT_NULL_URL };
};
