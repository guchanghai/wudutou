/**
 * Filters the given array of TranslatedDetail objects based on the provided previous tabs.
 * It returns a new array containing only the TranslatedDetail objects whose lang_code does not exist in the previous tabs.
 *
 * @param autodubArr - The array of TranslatedDetail objects to filter.
 * @param prevTabs - The array of previous tabs to compare against.
 * @returns The filtered array of TranslatedDetail objects.
 */
 
export const filterLanguages = (
  autodubArr: Array<TranslatedDetailModel>,
  prevTabs: Array<Tab>,
): Array<TranslatedDetailModel> => {
  return autodubArr.filter(
    (autodub: TranslatedDetailModel) =>
      !prevTabs.some((prevTab) => prevTab.id === autodub.lang_code),
  );
};
 
/**
 * Retrieves new tabs based on the autodub array, previous tabs, and supported languages map.
 * @param autodubArr - The array of TranslatedDetailModel objects representing autodub data.
 * @param prevTabs - The array of Tab objects representing previous tabs.
 * @param supportedLanguagesMap - The map of language codes to Language objects representing supported languages.
 * @returns A promise that resolves to an array of Tab objects.
 */
export const getNewTabs = (
  autodubArr: Array<TranslatedDetailModel>,
  prevTabs: Array<Tab>,
  supportedLanguagesMap: Map<string, Language>,
): Promise<Array<Tab>> => {
  // Function to filter the autodub array based on the previous tabs
  const filteredAutodubArr = filterLanguages(autodubArr, prevTabs);
 
  return Promise.all(
    filteredAutodubArr.map(
      async (
        {
          lang_code: langCode,
          text: transcriptUrl,
          path: mediaUrl,
        }: TranslatedDetailModel,
        index: number,
      ) => {
        let transcript: Array<Transcript> = [];
        if (transcriptUrl) {
          transcript = (await fetchLangAndTranscript(transcriptUrl)).transcript;
        }
 
        return {
          id: langCode,
          heading: supportedLanguagesMap.get(langCode)?.heading ?? langCode,
          url: mediaUrl,
          transcript,
          isSelected: index === 0,
        };
      },
    ),
  );
};
