      if (error) {
        showNegativeToast(error);
        return;
      }

      const shouldTranscodeInputMedia = autodubStore.shouldTranscodeMedia();
      try {
        setShowLoadingState(true);
        // Once upload starts, we don't need to poll for async queue.
        apiJobQueuePoller.reset();
        try {
          let uploadResponse;
          if (shouldTranscodeInputMedia) {
            // generate first thumbnail and upload only post transcoding input media
            uploadResponse = await acpChunkUpload(file);
          } else {
            const responses = await Promise.all([
              generateFirstThumbnailAndUpload(accessToken, file),
              acpChunkUpload(file),
            ]);
            uploadResponse = responses[1];
          }
          fileAccessInfo = {
            urlPath: uploadResponse.response,
            assetId: uploadResponse.assetId,
          };
        } catch (err) {
          // Analytics for upload failure event
          const errorDetails = {
            message: (err as Error).message,
            code: -1,
            type: undefined,
          };
          analytics.recordAutodubUploadFailureEvent(
            //TODO: update file input type - DVAGENAI-1960
            FeedbackUIViewType.FileBrowser,
            errorDetails.code,
            errorDetails.type,
            errorDetails.message,
          );
          // Analytics Event code ends here
          if (
            err instanceof ACPError &&
            err.getType() === QUOTA_EXCEEDED_ERROR
          ) {
            setNotEnoughStorage(true);
            return;
          }
          throw err;
        }
 
        if (shouldTranscodeInputMedia) {
          try {
            const fileUrl = await getTranscodedMediaUrl(
              fileAccessInfo,
              accessToken,
            );
            await Promise.all([
              autodubStore.updateTranscodedVideoState(fileUrl),
              generateFirstThumbnailAndUploadWithFileUrl(
                accessToken,
                fileUrl,
                isVideoFile(file),
              ),
            ]);
            // Update the file access info with the transcoded media url
            fileAccessInfo.urlPath = fileUrl;
          } catch (err) {
            publishErrorEvent({
              message: "Failed to transcode media.",
              error: err as Error,
            });
            showNegativeToast(ErrorCode.invalidFileType);
            return;
          }
        }
 
        autodubStore.setInputMediaAccessInfo(
          fileAccessInfo.urlPath,
          fileAccessInfo.assetId,
        );
 
        // Analytics for upload success event
        analytics.recordAutodubUploadSuccessEvent(
          //TODO: update file input type - DVAGENAI-1960
          FeedbackUIViewType.FileBrowser,
        );
        // Analytics Event code ends here
        onFileUploaded();
      } catch (err) {
        publishErrorEvent({
          message: "Failed to upload media.",
          error: err as Error,
        });
        showNegativeToast(ErrorCode.fileUploadFailure);
      } finally {
        setShowLoadingState(false);
      }
    };

    };

     * @param e - event on trigger of feedback.

      const detail = (e as CustomEvent).detail;
      const feedbackType = detail.feedbackType;
      let eventSubtype: SubmitFeedbackEventSubtype | undefined;
      switch (feedbackType) {
        case FeedbackEventValue.positive:
        case FeedbackEventValue.negative:
          eventSubtype = SubmitFeedbackEventSubtype.feedback;
          break;
        case FeedbackEventValue.report:
          eventSubtype = SubmitFeedbackEventSubtype.report;
          break;
        default:
          publishErrorEvent({
            message: `unknown feedback type: ${feedbackType}`,
          });
          return;
      }

      analytics.recordClickFeedback(
        FeedbackId.Autodub,
        feedbackType as ClickFeedbackEventSubtype,
        FeedbackUIViewType.Normal,
        -1, // seed
      );

        source: FeedbackUIViewType.Normal,
        seed: 0,
