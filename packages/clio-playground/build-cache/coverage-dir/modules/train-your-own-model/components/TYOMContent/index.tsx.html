import CCLibrary from "@spectrum-icons/workflow/CCLibrary";
import CloseCircle from "@spectrum-icons/workflow/CloseCircle";
import { Button } from "@swc-react/button";
import { DialogWrapper } from "@swc-react/dialog";
import { OverlayTrigger, type OverlayTriggerType } from "@swc-react/overlay";
import { ProgressCircle } from "@swc-react/progress-circle";
import { Textfield, type TextfieldType } from "@swc-react/textfield";
import {
  type ChangeEvent,
  type DragEvent,
  useCallback,
  useRef,
  useState,
} from "react";
import { useIntl } from "react-intl";
import { FileInput } from "shared/components/FileInput/FileInput";
import { Flex } from "shared/components/Flex/Flex";
import { IconButton } from "shared/components/IconButton/IconButton";
import { useSPLContext } from "shared/contexts/SPLContext";
import { ModuleId } from "shared/types/ModuleConfig";
 
import styles from "./css/TYOMContent.module.scss";
 
export default function TYOMContent() {
  const intl = useIntl();
  const { splService } = useSPLContext();
  const [files, setFiles] = useState<Array<File>>([]);
  const [name, setName] = useState("");
 
  const handleFileUpload = useCallback(
    (_: File, event: ChangeEvent<HTMLInputElement> | DragEvent) => {
      const newFiles =
        (event as ChangeEvent<HTMLInputElement>).target.files ??
        (event as DragEvent).dataTransfer.files;
 
      setFiles([...files, ...Array.from(newFiles)]);
    },
    [files, setFiles, splService],
  );
 
  const overlayTriggerRef = useRef<OverlayTriggerType>(null);
 
  const closeOverlay = useCallback(() => {
    if (!overlayTriggerRef.current) {
      return;
    }
    overlayTriggerRef.current.open = undefined;
  }, [overlayTriggerRef.current]);
 
  const submitStyle = useCallback(async () => {
    if (!name) {
      return;
    }
    void (await splService.tyom(ModuleId.Text2Image, Array.from(files), name));
    setFiles([]);
    closeOverlay();
  }, [name, files, setFiles, splService, closeOverlay]);
 
  const renderTrainingFiles = useCallback(() => {
    return (
      <Flex direction="row" wrap className={styles.filesContainer}>
        {files.map((file, index) => (
          <div className={styles.fileItem} key={index}>
            <img
              height={84}
              width={84}
              src={URL.createObjectURL(file)}
              className={styles.fileImage}
            />
            {/* Remove file */}
            <span
              className={styles.removeIcon}
              onClick={() => setFiles(files.filter((_file) => file !== _file))}
            >
              <CloseCircle size="S" />
            </span>
          </div>
        ))}
      </Flex>
    );
  }, [files, setFiles]);
 
  const renderStyleNameInput = useCallback(() => {
    return (
      <Textfield
        type="text"
        required
        placeholder={intl.formatMessage({
          id: "@clio/playground:tyom-style-name-placeholder",
          defaultMessage: "Name your style",
        })}
        input={(event) => setName((event.currentTarget as TextfieldType).value)}
      >
        {/* Prevent the default help text content from rendering */}
        <span slot="negative-help-text"></span>
        <span slot="help-text"></span>
        <span slot="pass-through-help-text-0"></span>
        <span slot="pass-through-help-text-1"></span>
      </Textfield>
    );
  }, [setName]);
 
  const renderSubmitButton = useCallback(() => {
    return (
      <OverlayTrigger
        ref={overlayTriggerRef}
        className={styles.submitButtonOverlayContainer}
        type="modal"
      >
        <Button
          slot="trigger"
          onClick={() => void submitStyle()}
          disabled={!name.length}
          variant="primary"
        >
          {intl.formatMessage({
            id: "@clio/playground:tyom-submit-button-label",
            defaultMessage: "Submit",
          })}
        </Button>
        <DialogWrapper
          slot="click-content"
          dismissable={false}
          underlay
          noDivider
        >
          <Flex direction="row">
            <ProgressCircle size="s" indeterminate />
            <span className={styles.uploadingDialogMessage}>
              {intl.formatMessage({
                id: "@clio/playground:tyom-uploading-images-message",
                defaultMessage:
                  "Uploading your images, please don't close this window...",
              })}
            </span>
          </Flex>
        </DialogWrapper>
      </OverlayTrigger>
    );
  }, [name, submitStyle, overlayTriggerRef]);
 
  const renderStyleTrainer = useCallback(() => {
    return (
      <Flex direction="column" align="center">
        <Flex direction="row" justify="space-between" width={300}>
          {renderStyleNameInput()}
          {renderSubmitButton()}
        </Flex>
        {renderTrainingFiles()}
      </Flex>
    );
  }, [renderTrainingFiles, name, setName]);
 
  const renderTrainFromLibrariesButton = useCallback(() => {
    return (
      <OverlayTrigger
        type="modal"
        className={styles.trainFromLibrariesOverlayContainer}
      >
        <IconButton
          slot="trigger"
          showAsButton
          label={intl.formatMessage({
            id: "@clio/playground:tyom-upload-select-library",
            defaultMessage: "Select Brand or Library",
          })}
          icon={<CCLibrary />}
          variant="primary"
          treatment="outline"
        />
        <DialogWrapper
          slot="click-content"
          headline={intl.formatMessage({
            id: "@clio/playground:tyom-upload-select-library-dialog-header",
            defaultMessage: "Your Brands & Libraries",
          })}
          dismissable
          underlay
          noDivider
        >
          {/* TODO: render library picker */}
          Coming soon...
        </DialogWrapper>
      </OverlayTrigger>
    );
  }, []);
 
  const renderFileInputSecondaryContent = useCallback(() => {
    if (files.length) {
      return renderStyleTrainer();
    }
    return (
      <Flex direction="column">
        {intl.formatMessage({
          id: "@clio/playground:tyom-upload-sub-heading",
          defaultMessage:
            "Or create a style based on one of your Brands & Libraries",
        })}
        {renderTrainFromLibrariesButton()}
      </Flex>
    );
  }, [files.length, renderStyleTrainer, renderTrainFromLibrariesButton]);
 
  return (
    <Flex
      className={styles.container}
      align="center"
      justify="start"
      direction="column"
    >
      <FileInput
        onSubmit={handleFileUpload}
        accept="image/jpeg,image/png"
        multiple
        buttonVariant="accent"
        buttonLabel={intl.formatMessage({
          id: "@clio/playground:tyom-upload-button-label",
          defaultMessage: "Upload images",
        })}
        title={{
          label: intl.formatMessage({
            id: "@clio/playground:tyom-title",
            defaultMessage: "Create a custom style",
          }),
          props: { marginTop: "size-500" },
        }}
        heading={{
          label: intl.formatMessage({
            id: "@clio/playground:tyom-instructions",
            defaultMessage:
              "Provide 15 or more images that best represent your visual style.",
          }),
          props: { marginTop: "size-100", marginBottom: "size-300" },
          fontSize: "var(--spectrum-font-size-300)",
        }}
        subHeading={{
          label: renderFileInputSecondaryContent(),
          props: { marginTop: "size-200", marginBottom: "size-500" },
        }}
        style={{ width: "100%" }}
      />
    </Flex>
  );
}
