import { useEffect, useRef, useState } from "react";
import { defineMessages, type MessageDescriptor, useIntl } from "react-intl";
import { ToastCopyLinkAndSaveError } from "shared/components/ToastCopyLinkAndSaveError/ToastCopyLinkAndSaveError";
import { useAnalyticsContext } from "shared/contexts/AnalyticsContext";

import { useShareSheetStoreContext } from "shared/contexts/ShareSheetContext";

import { DATA } from "shared/data/Data";
import {
  FPS_CREATE_ERROR,
  type StyleCreate,
} from "shared/document/DocumentTypes";

  createStyleKitItem,
  createStyleKitItemParams,
  getStyleKitErrorDescription,
} from "shared/document/DocumentUtils";
import { useEnabledFeatures } from "shared/hooks/useFeatureEnabled";

import {
  type FloodgateFlagName,

  FormContent,
  SectionFormContent,
  StyleKitFormData,
} from "../../types";

  checkForReservedNames,
  isValidAssetName,
} from "../../utils/text2ImageSharedUtils";

const messages = defineMessages({
  uniqueNameMessage: {
    id: "@clio/playground:style-kit-unique-name-help-text",
    defaultMessage: "Enter a unique name.",
  },
  hasSpecialCharactersMessage: {
    id: "@clio/playground:style-kit-special-characters-help-text",
    defaultMessage: "Names can't use special characters.",
  },
  endsWithPeriodMessage: {
    id: "@clio/playground:style-kit-ends-with-period-help-text",
    defaultMessage: "Names can't end with a period.",
  },
  hasControlCharactersMessage: {
    id: "@clio/playground:style-kit-control-characters-help-text",
    defaultMessage: "Names can't contain control characters.",
  },
  blockedWords: {
    id: "@clio/playground:style-kit-reserved-names-help-text",
    defaultMessage: "Names can't use blocked words.",
  },
  styleCreateErrorToastMessage: {
    id: "@clio/playground:style-kit-create-toast-message",
    defaultMessage: "Saving style kit failed. Please try again.",
  },
  toastMessage: {
    id: "@clio/playground:style-kit-toast-message",
    defaultMessage:
      "{length, select, long {Style kit {name}... saved} other {Style kit {name} saved} }",
  },
});
const StyleKitModal = observer(() => {
  const toast = useToastContext();
  const { openShareSheet, closeShareSheet } = useShareSheetStoreContext();
  const { styleKitModel, generateModel, predictedModel } = text2ImageStore;
  const { accessToken } = useAuthContext();
  const { analytics } = useAnalyticsContext();
  const { isFulfillableItemEnabled } = useEntitlementContext();
  const floodgateFlags: FloodgateFlagsMap | undefined = useEnabledFeatures();
  const hasFeatureFlag = (flag: FloodgateFlagName) => {
    return floodgateFlags?.[flag] ? true : false;
  };
  const hasOCAccess = () => {
    return isFulfillableItemEnabled("object_composites") &&
      hasFeatureFlag("firefly-object-composites")
      ? true
      : false;
  };
  const { text2ImageSidePanelGenerationStore } =
    useText2ImageSidebarStoreContext();
  const [helpText, setHelpText] = useState({
    invalid: false,
    message: undefined as MessageDescriptor | undefined,
  });
  const [loading, setLoading] = useState(false);
  const intl = useIntl();
 
  const { uiTemplateModel } = styleKitModel;
  const ref = useRef<UnderlayType>(null);
 
  const prompt = {
    key: "prompt",
    locked: uiTemplateModel.isLocked("prompt"),
  };
 
  const formData: StyleKitFormData = {
    prompt,
  };
 
  const createSection = (
    items: Array<Text2ImageTemplateModelLockedKeysType>,
  ) => {
    return items.map((key) => ({
      key,
      locked: uiTemplateModel.isLocked(key),
    }));
  };
 
  const general = createSection(["aspectRatio"]);
  const contentType = createSection(["contentType", "exif"]);
  const structure = createSection([
    "objectComposite",
    "structureReferenceImage",
    "structureAdherence",
  ]);
  const style = createSection([
    "rlhfCreativity",
    "guideStrength",
    "referenceImage",
    "styles",
    "colorAndTone",
    "lighting",
    "composition",
  ]);
  const sectionsWithFormContent: SectionFormContent = [
    { section: general, property: "general" },
    { section: contentType, property: "contentType" },
    { section: structure, property: "structure" },
    { section: style, property: "style" },
  ];
 
  // "prompt" should be put in the array since it is not in any category
  const visibleProperties: Array<Text2ImageStyleKitModelKeyType> = ["prompt"];
  sectionsWithFormContent.forEach((item) => {
    const filteredItems = item.section.filter(
      (panelItem) =>
        !text2ImageSidePanelGenerationStore.isLocked(panelItem.key),
    );
 
    if (filteredItems.length <= 0) {
      return;
    }
 
    if (item.property in formData) {
      (formData[item.property] as Array<FormContent>).push(...filteredItems);
    } else {
      formData[item.property] = [...filteredItems];
    }
 
    visibleProperties.push(
      ...filteredItems.map<Text2ImageStyleKitModelKeyType>(
        (i) => i.key as Text2ImageStyleKitModelKeyType,
      ),
    );
  });
 
  useOnUnmount(() => {
    closeShareSheet();
  });
 
  useEffect(() => {
    const disposer = reaction(
      () => styleKitModel.isOpen,
      (isOpen) => {
        const isClosed = !isOpen;
        if (isClosed) {
          return;
        }
        styleKitModel.initialize();
      },
    );
    return () => disposer();
  }, []);
 
  useEffect(() => {
    const disposer = reaction(
      () => [styleKitModel.name],
      () => {
        const name = styleKitModel.name.trim();
        const isValidName = isValidAssetName(name);
        const hasReservedNames = checkForReservedNames(name);
 
        if (!isValidName) {
          setHelpText({
            invalid: true,
            message: messages.hasSpecialCharactersMessage,
          });
        } else if (hasReservedNames) {
          setHelpText({
            invalid: true,
            message: messages.blockedWords,
          });
        } else {
          setHelpText({
            invalid: false,
            message: undefined,
          });
        }
      },
    );
 
    return () => {
      disposer();
    };
  }, []);
 
  // check if all categories are not available then do not return the category section otherwise return category with the available section
  const categoryOpen = (val: boolean) => {
    const category: Record<string, boolean> = {};
    Object.keys(formData).forEach((key) => {
      if (key !== "name") {
        category[key] = val;
      }
    });
    return category;
  };
 
  const [categoryIsOpen, setCategoryIsOpen] = useState(categoryOpen(true));
 
  // check if all properties are unlocked (thus remixable)
  const checkAllUnLocked = (formData: StyleKitFormData) => {
    const formContents = Object.values(formData);
    return formContents.every((formContent) => {
      if (Array.isArray(formContent)) {
        return formContent.every((content) => !content.locked);
      } else {
        return !formContent.locked;
      }
    });
  };
 
  const isSelectAllChecked = checkAllUnLocked(formData);
  const disableSaveBtn = helpText.invalid || styleKitModel.name.trim() === "";
 
  const unlockAllVisibleProperties = () => {
    visibleProperties.forEach((prop) => {
      styleKitModel.unlockProperty(prop as keyof Text2ImageEditModel);
    });
  };
 
  const lockAllVisibleProperties = () => {
    visibleProperties.forEach((prop) => {
      styleKitModel.lockProperty(prop as keyof Text2ImageEditModel);
    });
  };
 
  const updateAllProperties = (e: Event) => {
    const checkbox = e.target as HTMLInputElement;
    if (checkbox.checked) {
      unlockAllVisibleProperties();
    } else {
      lockAllVisibleProperties();
      analytics.recordClickSelectAllStyleKit(FeedbackId.Text2Image);
    }
  };
 
  const handleOpenCategory = (
    category: keyof StyleKitFormData,
    isOpen: boolean,
  ) => {
    setCategoryIsOpen({ ...categoryIsOpen, [category]: isOpen });
  };
 
  const resetFormData = () => {
    styleKitModel.setName("");
    unlockAllVisibleProperties();
  };
 
  const updateCheckboxWithSubCategory = (
    e: Event,
    category: keyof StyleKitFormData,
  ) => {
    const options = formData[category] as Array<FormContent>;
 
    options.forEach((item: FormContent) => {
      styleKitModel.handleUpdateProperty(
        e,
        item.key as keyof Text2ImageEditModel,
      );
    });
  };
 
  const handleUpdateProperty = (e: Event, key: keyof Text2ImageEditModel) => {
    styleKitModel.handleUpdateProperty(e, key);
  };
 
  const handleCancel = () => {
    resetFormData();
    setCategoryIsOpen(categoryOpen(true));
    setHelpText({
      invalid: false,
      message: undefined,
    });
    styleKitModel.open(false);
 
    analytics.recordClickCancelSaveStyleKit(FeedbackId.Text2Image);
  };
 
  const handleCopyLink = (
    accessType: ShareAccessType,
    invitedUsers: Set<string>,
  ) => {
    const unlockedItems = getUnlockedItems(formData);
 
    analytics.recordSuccessShareStyleKit(
      FeedbackId.Text2Image,
      invitedUsers.size,
      accessType,
      {
        unlockedItems,
        unlockedItemsCount: unlockedItems.length,
      },
    );
  };
 
  const handleInviteFulfilled = (invitedUsers: Set<string>) => {
    analytics.recordClickInviteShareStyleKit(
      FeedbackId.Text2Image,
      invitedUsers.size,
    );
  };
 
  const resetEditModelWithStyleKit = (model: Text2ImageEditModel) => {
    const updateModel = {
      ...model,
      previewSeed: undefined,
    };
    text2ImageStore.setEditModel(DATA.normalize(updateModel));
  };
 
  const loadStyleKit = (assetId: string) => {
    const styleKitLoader = new StyleKitsLoader(
      authStore,
      documentStore,
      customModelStore,
      hasOCAccess(),
    );
 
    styleKitLoader
      .loadStyleKitsWithCaching(assetId)
      .then((model) => {
        resetEditModelWithStyleKit(model);
      })
      .catch((err) => {
        console.warn(err);
      });
    return;
  };
 
  const handleSave = async () => {
    const unlockedItems = getUnlockedItems(formData);
    const styleKitAnalyticsInfo = {
      unlockedItems,
      unlockedItemsCount: unlockedItems.length,
    };
    if (!helpText.invalid) {
      try {
        setLoading(true);
        if (!styleKitModel.seed) {
          throw new Error(
            "Can't save stylekit template, selected seed is undefined",
          );
        }
 
        analytics.recordClickSaveStyleKit(FeedbackId.Text2Image);
 
        if (
          !styleKitModel.determineLockedStatesForInvisibleProperties(
            visibleProperties,
            generateModel,
          )
        ) {
          throw Error(
            "Error determining locked states for invisible properties",
          );
        }
 
        const selectedSeed = styleKitModel.seed;
        const createStyleKitItemData = (await createStyleKitItemParams({
          url: styleKitModel.url,
          generateModel: {
            ...generateModel,
            previewSeed: selectedSeed,
          },
          seed: selectedSeed,
          templateModel: styleKitModel.templateModel,
        })) as StyleCreate;
 
        const { assetId } = await createStyleKitItem(
          createStyleKitItemData,
          styleKitModel.name,
          accessToken,
          predictedModel ?? {},
        );
        const assetName = styleKitModel.name;
        styleKitModel.open(false);
        setCategoryIsOpen(categoryOpen(true));
        toast.show(
          <StyleKitToastViewInFileBtn
            toastMessage={messages.toastMessage}
            fileName={styleKitModel.name}
          />,
          {
            variant: ToastVariant.POSITIVE,
            messageType: MessageType.INFO,
          },
        );
 
        analytics.recordSuccessSaveStyleKit(
          FeedbackId.Text2Image,
          styleKitAnalyticsInfo,
        );
 
        // set isDirty Flag for styleKit Updater if the creation is success, to refresh the listing
        documentStore.styleKits.setIsDirtyFlag(true);
        loadStyleKit(assetId);
        openShareSheet(
          assetId,
          assetName,
          handleCopyLink,
          handleInviteFulfilled,
        );
        resetFormData();
      } catch (error) {
        const errMessage = (error as Error).message;
        analytics.recordErrorSaveStyleKit(
          FeedbackId.Text2Image,
          styleKitAnalyticsInfo,
          errMessage,
          getStyleKitErrorDescription(errMessage),
        );
        if (errMessage === FPS_CREATE_ERROR.CONFLICT) {
          setHelpText({
            invalid: true,
            message: messages.uniqueNameMessage,
          });
        } else if (errMessage === FPS_CREATE_ERROR.USER_NOT_ENTITLED) {
          toast.show(
            <ToastCopyLinkAndSaveError
              errorCode={PublicShareErrorCode.STORAGE_NOT_ASSIGNED}
            />,
            {
              variant: ToastVariant.NEGATIVE,
            },
          );
        } else {
          toast.show(
            intl.formatMessage(messages.styleCreateErrorToastMessage),
            {
              variant: ToastVariant.NEGATIVE,
            },
          );
        }
      } finally {
        setLoading(false);
      }
    }
  };
 
  useEffect(() => {
    if (!ref.current) {
      return;
    }
    const handleCloseOnEscape = (e: KeyboardEvent) => {
      if (e.key === "Escape") {
        handleCancel();
      }
    };
    const el = ref.current;
    el.addEventListener("click", handleCancel);
    document.body.addEventListener("keydown", handleCloseOnEscape);
    return () => {
      document.body.removeEventListener("keydown", handleCloseOnEscape);
    };
  }, [ref.current !== null]);
 
  return (
    <StyleKitModalContainer
      disableSaveBtn={disableSaveBtn}
      loading={loading}
      styleKitModel={styleKitModel}
      formData={formData}
      categoryIsOpen={categoryIsOpen}
      updateAllProperties={updateAllProperties}
      handleOpenCategory={handleOpenCategory}
      isSelectAllChecked={isSelectAllChecked}
      updateCheckboxWithSubCategory={updateCheckboxWithSubCategory}
      handleUpdateProperty={handleUpdateProperty}
      handleCancel={handleCancel}
      handleSave={handleSave}
      helpText={helpText}
      ref={ref}
    />
  );
});
 
export default StyleKitModal;
