enum ObjectCompositeErrorType {
  SMALL_IMAGE,
  GENERIC,
}
class ObjectCompositeError extends Error {
  constructor(private _errorType: ObjectCompositeErrorType, message: string) {
    super(message);
  }
  get errorType() {
    return this._errorType;
  }
}

const getPlacementOrDefault = (editModel: Text2ImageModelStore) => {
  return (
    editModel.objectComposite?.placement ?? DEFAULT_THUMBNAIL_OBJECT_PLACEMENT
  );
};
 
const useSetPromptBarImageUrl = () => {
  const [promptBarImageUrl, setPromptBarImageUrlInternal] = useState<string>();
  const setPromptBarImageUrl = async (blobRef?: BlobRef) => {
    blobstore.returnUrl(promptBarImageUrl);
    if (blobRef) {
      setPromptBarImageUrlInternal(await blobstore.borrowUrl(blobRef));
    } else {
      setPromptBarImageUrlInternal(undefined);
    }
  };
  return { promptBarImageUrl, setPromptBarImageUrl };
};
 
const ObjectReferenceUpload = ({
  isInspirationPage,
  isSideBar,
  handleResetClick,
}: Props) => {
  const { editModel } = text2ImageStore;
  const fileInput = useRef<HTMLInputElement>(null);
  const { analytics } = useAnalyticsContext();
  const [heroImage, setHeroImage] = useState<ReferenceImageSchema | undefined>(
    editModel.objectComposite?.heroImage,
  );
  const { promptBarImageUrl, setPromptBarImageUrl } = useSetPromptBarImageUrl();
 
  const [heroImageSize, setHeroImageSize] = useState<
    ObjectCompositonImageSize | undefined
  >(editModel.objectComposite?.heroImageSize);
  const [placement, setPlacement] = useState<ObjectCompositionPlacement>(
    getPlacementOrDefault(editModel),
  );
  const intl = useIntl();
  const isMobile = useIsMobile();
  const [showObjectDialog, setShowObjectDialog] = useState<objectDialogTypes>(
    objectDialogTypes.none,
  );
  const {
    OBJECT_COMPOSITE_PROMPTBAR_COACHMARK_KEY,
    OBJECT_COMPOSITE_PROMPTBAR_UPLOAD_COACHMARK_KEY,
  } = UDS_KEYS;
  const [showObjectUploadCoachmark, setShowObjectUploadCoachmark] =
    useState<boolean>(true);
  const [showPromptBarCoachmark, setShowPromptBarCoachmark] =
    useState<boolean>(false);
  const { handleObjectCompositeUpload, uploadObjectComposite } =
    useObjectCompositeModalHandler();
 
  const { confirmationAccepted } = useUserSettingStoreContext();
 
  const errorMessageStore = useInstanceRef(PositionModalErrorMessageStore);
 
  const cancelDialog = () => {
    setHeroImage(editModel.objectComposite?.heroImage);
    setHeroImageSize(editModel.objectComposite?.heroImageSize);
    setPlacement(getPlacementOrDefault(editModel));
    setShowObjectDialog(objectDialogTypes.none);
    analytics.recordClickCancelObjectComposite();
  };
 
  const confirmDialog = () => {
    setShowObjectDialog(objectDialogTypes.none);
    setShowPromptBarCoachmark(true);
    if (heroImage && heroImageSize && placement) {
      uploadObjectComposite(heroImage, heroImageSize, placement);
    }
    analytics.recordClickConfirmObjectComposite();
  };
 
  useEffect(() => {
    if (isSideBar) {
      return;
    }
    analytics.recordRenderObjectCompositeInPromptBar();
  }, []);
 
  const fetchImageUrl = async (heroImage: ReferenceImageSchema | undefined) => {
    if (heroImage) {
      await setPromptBarImageUrl(heroImage.ref);
    } else {
      await setPromptBarImageUrl(undefined);
    }
  };
 
  useEffect(() => {
    void fetchImageUrl(editModel.objectComposite?.heroImage);
  }, [editModel.objectComposite?.heroImage]);
 
  const handleFileUpload = async (file: Blob) => {
    try {
      const resizedImageResults = await handleObjectCompositeUpload(file);
      if (isBothSizeLessThan(resizedImageResults.size, IMAGE_MIN_DIMENSION)) {
        if (!heroImage) {
          // if no image has been uploaded before, we should display an error dialog for a small image
          throw new ObjectCompositeError(
            ObjectCompositeErrorType.SMALL_IMAGE,
            "Small image",
          );
        }
        errorMessageStore.current.displayErrorMessage(
          {
            title: intl.formatMessage(objectImageMessages.smallImageTitle),
            message: intl.formatMessage(
              objectImageMessages.smallImageUploadedMessage,
            ),
          },
          ERROR_DISPLAY_TIME_IN_MILLISECONDS,
        );
        return;
      }
 
      const blobReference = await blobstore.getRef(resizedImageResults.blob);
      if (blobReference) {
        const referenceImageId = isLocalBlobRef(blobReference)
          ? blobReference.localBlobRef
          : blobReference.remoteBlobRef;
        const uploadedImage: ReferenceImageSchema = {
          type: "object-reference",
          id: referenceImageId,
          ref: blobReference,
        };
 
        setHeroImage(uploadedImage);
        setHeroImageSize(resizedImageResults.size);
        setShowObjectDialog(objectDialogTypes.position);
      }
    } catch (e) {
      console.error("Error thrown at file upload", e);
 
      // According to [FIREFLY-22325], we need reset the heroImage and heroImageSize when error thrown
      setHeroImage(editModel.objectComposite?.heroImage);
      setHeroImageSize(editModel.objectComposite?.heroImageSize);
 
      if (
        e instanceof ObjectCompositeError &&
        e.errorType === ObjectCompositeErrorType.SMALL_IMAGE
      ) {
        setShowObjectDialog(objectDialogTypes.uploadSmallImageError);
      } else {
        setShowObjectDialog(objectDialogTypes.uploadGenericError);
      }
    }
  };
 
  const handleDrop = async (event: DragEvent): Promise<void> => {
    setShowObjectUploadCoachmark(false);
    if (event.dataTransfer.files.length > 0) {
      const file = event.dataTransfer.files.item(0);
      if (file) {
        event.preventDefault();
        await handleFileUpload(file);
      } else {
        throw new Error("No image source");
      }
    }
  };
 
  const handleFileSubmit = async (): Promise<void> => {
    try {
      setShowObjectUploadCoachmark(false);
      if (fileInput.current?.files && fileInput.current.files.length > 0) {
        const file = fileInput.current.files[0];
        if (file.type && SUPPORTED_IMAGE_MIME_TYPE.includes(file.type)) {
          try {
            await handleFileUpload(file);
          } catch (e) {
            console.error("Error thrown at file upload", e);
          }
        }
        fileInput.current.value = "";
      } else {
        console.error("No image source");
      }
    } catch (e) {
      console.error("No image source");
    }
  };
  const coachmarkStyle: CSSProperties = {
    position: "relative",
    bottom: "var(--spectrum-global-dimension-size-2500)",
    left: `var(--spectrum-global-dimension-size-3600)`,
  };
 
  const coachmarkStylePrompt: CSSProperties = {
    position: "relative",
    bottom: "var(--spectrum-global-dimension-size-2500)",
    left: "410px",
  };
 
  const uploadButtonHeader = () => {
    return (
      <Flex
        direction="row"
        gap="size-150"
        className={styles.heading}
        data-testid="object-reference-upload-model-header"
      >
        <div className={styles.ocButtonTitle}>
          {intl.formatMessage(objectImageMessages.ocTitle)}
        </div>
        <div>
          <InfoButton
            toolTiplabel={intl.formatMessage(
              objectImageMessages.objectImageTooltip,
            )}
            placement="top"
          />
        </div>
      </Flex>
    );
  };
 
  const onClickRemove = async () => {
    await setPromptBarImageUrl(undefined);
    actions.clearObjectComposition();
    analytics.recordClickClearStyleObjectComposite();
  };
 
  const onClickImage = () => {
    setShowObjectDialog(objectDialogTypes.position);
    setHeroImageSize(editModel.objectComposite?.heroImageSize);
    setPlacement(getPlacementOrDefault(editModel));
  };
 
  return (
    <div>
      <input
        type="file"
        ref={fileInput}
        onChange={() => void handleFileSubmit()}
        style={{ display: "none" }}
        accept={SUPPORTED_IMAGE_MIME_TYPE.join(",")}
      />
      <div>
        {showObjectDialog === objectDialogTypes.uploadGenericError && (
          <ObjectCompositeErrorImageModal
            onCancel={() => setShowObjectDialog(objectDialogTypes.none)}
            handleFileUpload={(file) => {
              void handleFileUpload(file);
            }}
            content={intl.formatMessage(
              objectImageMessages.errorDialogContent,
              {
                a: (chunks) => (
                  <a
                    href={intl.formatMessage(objectImageMessages.userGuideUrl)}
                    target="_blank"
                  >
                    {chunks}
                  </a>
                ),
                icon: <svg />,
              },
            )}
          />
        )}
        {showObjectDialog === objectDialogTypes.uploadSmallImageError && (
          <ObjectCompositeErrorImageModal
            onCancel={() => setShowObjectDialog(objectDialogTypes.none)}
            handleFileUpload={(file) => {
              void handleFileUpload(file);
            }}
            content={intl.formatMessage(
              objectImageMessages.smallImageErrorDialogContent,
              { minSize: IMAGE_MIN_DIMENSION },
            )}
          />
        )}
 
        {showObjectDialog === objectDialogTypes.position &&
          !isMobile &&
          heroImage && (
            <ObjectCompositePositionModal
              onCancel={cancelDialog}
              onConfirm={confirmDialog}
              setPlacement={setPlacement}
              placement={placement}
              heroImageSize={heroImageSize}
              heroImage={heroImage}
              handleFileUpload={(file) => {
                void handleFileUpload(file);
              }}
              errorMessageStore={errorMessageStore.current}
            />
          )}
 
        {showObjectUploadCoachmark &&
          isInspirationPage &&
          !confirmationAccepted(
            OBJECT_COMPOSITE_PROMPTBAR_UPLOAD_COACHMARK_KEY,
          ) && (
            <Text2ImageStaticCoachmark
              style={coachmarkStyle}
              udsKey={OBJECT_COMPOSITE_PROMPTBAR_UPLOAD_COACHMARK_KEY}
              messages={{
                title: objectImageMessages.coachmarkTitleForObjectUpload,
                content: objectImageMessages.coachmarkContentForObjectUpload,
              }}
              analyticsType={Text2ImageCoachmarkTypes.ObjectComposite}
            />
          )}
        {}
        {showPromptBarCoachmark &&
          isInspirationPage &&
          !confirmationAccepted(OBJECT_COMPOSITE_PROMPTBAR_COACHMARK_KEY) && (
            <Text2ImageStaticCoachmark
              style={coachmarkStylePrompt}
              udsKey={OBJECT_COMPOSITE_PROMPTBAR_COACHMARK_KEY}
              messages={{
                title: objectImageMessages.coachmarkTitleForPromptbar,
                content: objectImageMessages.coachmarkContentForPromptbar,
              }}
              analyticsType={Text2ImageCoachmarkTypes.ObjectComposite}
            />
          )}
        {editModel.objectComposite?.heroImage && !isSideBar && (
          <>
            {uploadButtonHeader()}
            <div className={styles["reference-image"]}>
              <ImagePreview
                src={promptBarImageUrl}
                onClickRemove={onClickRemove}
                onClick={onClickImage}
                showRemoveButton
                showBorder
                width={uploadOCImageDimensions.width}
                height={uploadOCImageDimensions.height}
                label={intl.formatMessage(objectImageMessages.imagAlt)}
                removeLabel={intl.formatMessage(
                  objectImageMessages.DeleteButton,
                )}
                data-testid="prompt-bar-object-preview-image-box"
                close-button-testid="prompt-bar-object-close-button"
              />
            </div>
          </>
        )}
 
        {!editModel.objectComposite?.heroImage && !isSideBar && (
          <>
            {uploadButtonHeader()}
            <Dropzone
              onClick={() => fileInput.current?.click()}
              onDropCapture={(e) => void handleDrop(e)}
              className={styles.dropzone}
            >
              <div
                className={styles.ImageIconBox}
                data-testid="object-composition-upload-button"
              >
                <Flex direction="column" gap="size-50" align="center">
                  <div className={styles.objectImage}>
                    <UploadToCloudIcon
                      label={intl.formatMessage(
                        objectImageMessages.changeImage,
                      )}
                    />
                  </div>
                  <div className={styles.ocUploadTitle}>
                    {intl.formatMessage(
                      objectImageMessages.ocUploadButtonTitle,
                    )}
                  </div>
                </Flex>
              </div>
            </Dropzone>
          </>
        )}
        {isSideBar && (
          <SidebarInputImage
            ref={fileInput}
            imageReferenceUrl={promptBarImageUrl}
            label={intl.formatMessage(objectImageMessages.ocTitle)}
            additionalButtonLabel={intl.formatMessage(
              objectImageMessages.ocResetButtonTitle,
            )}
            showAdditionalButton={true}
            imagePreviewBoxTestId="structure-object-preview-box"
            uploadButtonTestId="structure-object-upload-image-button"
            browseButtonTestId="structure-object-reset-button"
            closeButtonTestId="structure-object-close-button"
            applyObjectFit
            onSubmit={() => void handleFileSubmit()}
            onClickRemove={() => void onClickRemove()}
            onClickImage={onClickImage}
            onUploadButtonClick={() => fileInput.current?.click()}
            onDropCapture={(event) => void handleDrop(event)}
            onAdditionalButtonClick={() => {
              handleResetClick?.();
            }}
            additionalButtonEnabled={
              text2ImageStore.hasObjectCompositionChanged
            }
            isObjectCompositeInputImage={true}
          />
        )}
      </div>
    </div>
  );
};
export default observer(ObjectReferenceUpload);
