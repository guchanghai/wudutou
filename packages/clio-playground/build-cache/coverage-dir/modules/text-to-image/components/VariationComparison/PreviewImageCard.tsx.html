import classNames from "classnames";
import { useState } from "react";
import { useObservable } from "shared/centralDispatch/hooks";
import { fromObservables } from "shared/centralDispatch/observable";
import { Thumbnail } from "shared/components/Thumbnail/Thumbnail";
import { ThumbUp } from "shared/icons/spectrum2Icons";
import { FeedbackUIViewType } from "shared/types/AnalyticsTypes";
import { type ImageAspectRatio } from "shared/types/ImageAspectRatio";
import { useTimeout } from "usehooks-ts";
 
import { observables } from "../../centralDispatch/observables.js";
import styles from "./css/VariationComparison.module.scss";
 
interface PreviewImageCardProps {
  className?: string;
  hideLabel?: boolean;
  id: string;
  isSelected: boolean;
  onClick: (key: string) => void;
  previewMode?: boolean;
  seed: number;
  url: string;
  showSelectedOverlay?: boolean;
  label: string;
  loading?: boolean;
  error?: Error;
  onRetry?: () => void;
  style?: React.CSSProperties;
}
 
type Model = {
  generatedPrompt: string;
  generatedAspectRatio: ImageAspectRatio;
};
 
const CardOverlayLike = () => {
  const [showOverlayAnimation, setShowOverlayAnimation] = useState(true);
 
  useTimeout(() => setShowOverlayAnimation(false), 1000);
 
  return (
    <div className={styles["selected-card-overlay"]}>
      <div
        className={classNames(styles["like-wrapper"], {
          [styles["like-wrapper-fade-out"]]: !showOverlayAnimation,
        })}
      >
        <ThumbUp />
      </div>
    </div>
  );
};
 
//  only exported for testing.
export const PreviewImageCardUI = ({
  id,
  url,
  seed,
  onClick,
  isSelected,
  hideLabel = false,
  previewMode = false,
  label,
  showSelectedOverlay = true,
  className = "",
  loading = false,
  error,
  onRetry,
  style,
  model,
}: PreviewImageCardProps & { model: Model }) => {
  const Title = () => {
    if (hideLabel) {
      return null;
    }
 
    if (previewMode) {
      return <h2>{label}</h2>;
    }
 
    return <h4>{label}</h4>;
  };
 
  return (
    <div className={styles["card-container"]}>
      <Title />
      <Thumbnail
        style={style}
        key={id}
        src={url}
        loading={loading}
        error={error}
        className={classNames(className, {
          [styles["avoid-clicks"]]: previewMode,
        })}
        alt={model.generatedPrompt}
        feedbackSource={FeedbackUIViewType["1Up"]}
        seed={seed}
        aspectRatio={model.generatedAspectRatio}
        hideToolbar
        selected={isSelected}
        onClick={() => onClick(id)}
        selectedColor={previewMode ? "none" : undefined}
        onRetry={onRetry}
      />
 
      {isSelected && showSelectedOverlay && <CardOverlayLike />}
    </div>
  );
};
 
//  Create the observable model that contains the business logic binding state and actions to the UI.
//  In this example we can create it statically and it contains no actions as this component is read only.
const observableModel = fromObservables({
  generatedAspectRatio: observables.generatedAspectRatio,
  generatedPrompt: observables.generatedPrompt,
});
 
export const PreviewImageCard = (props: PreviewImageCardProps) => {
  //  The useObservable will return the current observed value.
  const model = useObservable(observableModel);
  //  The returned model object will be undefined until all required observables are available.
  //  This means a UI component can be precise about what dependencies it has
  //  and will render as soon as all required dependencies are available.
  return model ? <PreviewImageCardUI {...props} model={model} /> : <></>;
};
