import { type Position, type Rect, rectToBounds } from "shared/types/Geometry";
 
export const HANDLE_CORNER_RADIUS = 8;
export const HANDLE_LINE_WIDTH = 2;
// Helper function to get the positions of the corners
export const getCorners = (rect: Rect): Array<Position> => {
  const bounds = rectToBounds(rect);
  return [
    { x: bounds.left, y: bounds.top }, // Top-left
    { x: bounds.right, y: bounds.top }, // Top-right
    { x: bounds.right, y: bounds.bottom }, // Bottom-right
    { x: bounds.left, y: bounds.bottom }, // Bottom-left
  ];
};
 
// Function to draw the border with corner circles
export const drawBorderAndHandles = (
  ctx: CanvasRenderingContext2D,
  rect: Rect,
  scaleFactor: number,
) => {
  const lineWidth = HANDLE_LINE_WIDTH * scaleFactor;
  const handleRadius = HANDLE_CORNER_RADIUS * scaleFactor;
  // Draw the border
  ctx.strokeStyle = "blue";
  ctx.lineWidth = lineWidth;
  ctx.strokeRect(rect.x, rect.y, rect.width, rect.height);
 
  // Draw the corner circles
  const corners = getCorners(rect);
  ctx.fillStyle = "white";
  ctx.strokeStyle = "blue";
  corners.forEach((corner: { x: number; y: number }) => {
    ctx.beginPath();
    ctx.arc(corner.x, corner.y, handleRadius, 0, 2 * Math.PI);
    ctx.fill();
    ctx.stroke();
  });
};
 
// Function to draw grid lines
export const drawGrid = (
  ctx: CanvasRenderingContext2D,
  gridSize: number, // Size of each grid cell
) => {
  const width = ctx.canvas.width;
  const height = ctx.canvas.height;
 
  // Set the style for the grid lines
  ctx.strokeStyle = "#ccc"; // Light grey color for the grid lines
  ctx.lineWidth = 1;
 
  // Draw the vertical grid lines
  for (let x = gridSize; x < width; x += gridSize) {
    ctx.beginPath();
    ctx.moveTo(x, 0);
    ctx.lineTo(x, height);
    ctx.stroke();
  }
 
  // Draw the horizontal grid lines
  for (let y = gridSize; y < height; y += gridSize) {
    ctx.beginPath();
    ctx.moveTo(0, y);
    ctx.lineTo(width, y);
    ctx.stroke();
  }
};
