// Calling createBrowserRouter too early may cause some issues if loaders are in the mix in
// the router configuration. We create a single global to avoid multiple instances when/if
// our Router is ever re-rendered due to upstream providers.
let browserRouterInstance: ReturnType<typeof createBrowserRouter> | undefined;
 
function Router() {
  const isMobileOrTable = useIsMobile() || useIsTablet();
  if (!browserRouterInstance) {
    browserRouterInstance = createBrowserRouter(
      createRoutesConfig(isMobileOrTable),
      {
        basename: ROUTER_BASE_NAME ?? undefined,
      },
    );
  }
  return <RouterProvider router={browserRouterInstance} />;
}
 
export default Router;
