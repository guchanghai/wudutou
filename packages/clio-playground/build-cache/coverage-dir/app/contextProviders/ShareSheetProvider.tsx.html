import { observer } from "mobx-react-lite";
import { useEffect, useMemo, useState } from "react";
import { isProdBuild } from "shared/config/BuildMode";
import { useAuthContext } from "shared/contexts/IMSContext";
import { usePrivacyContext } from "shared/contexts/PrivacyContext";
import { ShareSheetContext } from "shared/contexts/ShareSheetContext";
import { useThemeContext } from "shared/contexts/ThemeContext";
import { getLocaleFromSetting } from "shared/services/asset/CDNAssetUtils";
import { componentLoader } from "shared/services/component-loader/ComponentLoader";
import { ACP_API_KEY } from "shared/services/config/ACPConfig";
import { INGEST_ENV_COM_NAME, version } from "shared/services/ingest/Ingest";
import { ShareSheetStore } from "shared/stores/ShareSheetStore";
import { PrivacyGroup } from "shared/types/PrivacyTypes";
 
const shareSheetStore = new ShareSheetStore();
 
function _ShareSheetProvider({ children }: React.PropsWithChildren) {
  const { accessToken, config, userProfile } = useAuthContext();
  const { privacyGroups } = usePrivacyContext();
  const { colorTheme } = useThemeContext();
  const [loaded, setLoaded] = useState(false);
 
  const analyticsEnabled = useMemo(
    () => privacyGroups.includes(PrivacyGroup.performance),
    [privacyGroups],
  );
 
  const locale = getLocaleFromSetting();
 
  const shareSheetOptions = useMemo(
    () =>
      ({
        apiKey: ACP_API_KEY,
        analyticsEnabled,
        analyticsData: {
          source: {
            client_id: config?.clientId, // eslint-disable-line camelcase -- ingest object
            name: INGEST_ENV_COM_NAME,
            platform: "WEB",
            version: version,
          },
          event: {
            category: "WEB",
          },
        },
        canComment: false, // To make invitees have no comment access, as well as LinkAccess to show 'can view' instead of 'can comment'
        enableFacepile: true,
        environment: isProdBuild() ? "prod" : "stage",
        features: {
          groups: true,
        },
        imsData: {
          email: userProfile?.email,
          id: userProfile?.userId as string,
          token: accessToken,
          userId: userProfile?.userId,
        },
        inviteApiVersion: "V4",
        locale,
        openedDialog: "share",
        theme: colorTheme === "dark" ? "darkest" : colorTheme,
      } as any), // eslint-disable-line @typescript-eslint/no-explicit-any -- outdated Options type. See https://git.corp.adobe.com/CCX/ccx-sharing/blob/main/packages/component-web/README.md#configurable-options
    [
      accessToken,
      analyticsEnabled,
      colorTheme,
      config?.clientId,
      locale,
      userProfile?.email,
      userProfile?.userId,
      version,
    ],
  );
 
  // Load ShareSheet once
  useEffect(() => {
    if (shareSheetOptions.imsData?.token) {
      if (!loaded) {
        componentLoader.load("ShareSheet", "30.14.0", shareSheetOptions);
        setLoaded(true);
      } else {
        // TODO: ShareSheet's updateConfig overwrites the entire config object,
        // so we need to send the entire config object every time we update it.
        // This should be fixed in ShareSheet so that the config is merged with previous config
        // or selectively updates the theme and locale like Libraries does:
        // https://git.corp.adobe.com/ProjectCentral/cc-libraries-panel/blob/412520e2a044682bbe96db7c0d18b32c7acda184/app/js/environments/CCWebEnvironment.js#L967-L979
        componentLoader.updateConfig(shareSheetOptions);
      }
    }
  }, [shareSheetOptions]);
 
  return (
    <ShareSheetContext.Provider value={shareSheetStore}>
      {children}
    </ShareSheetContext.Provider>
  );
}
 
export const ShareSheetProvider = observer(_ShareSheetProvider);
