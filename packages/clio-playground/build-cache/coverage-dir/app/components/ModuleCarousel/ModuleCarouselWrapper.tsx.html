const GAP = 15; // should match css
 
const CLONED_ITEM_COUNT = 6;
 
const ModuleCarouselWrapperComponent = ({
  infinite = false,
}: ModuleCarouselWrapperProps) => {
  const intl = useIntl();
  const { isFeatureEnabled, floodgateFlagsWithOverrides } =
    personalizationStore;
  const isMobile = useIsMobile();
  const isMobileOrTablet = isMobile || useIsTablet();
  const [activeModuleId, setActiveModuleId] = useState<string>("");
  const { detectModuleStatus } = useModuleUtils();
  const { isFulfillableItemEnabled, isK12 } = useEntitlementContext();
  const { isNonAdobeEnterpriseUser } = useAuthContext();
  const isAutoScrollEnabled = isFeatureEnabled("firefly-homepage-auto-scroll");
  const isFeedbackEnabled = isFeatureEnabled("firefly-clio-feedback-module");
 
  const showFeedbackModule =
    isFeedbackEnabled &&
    !isK12 &&
    getLocaleFromSetting() === DEFAULT_LOCALE &&
    !isNonAdobeEnterpriseUser;
 
  const carouselInner = useRef<HTMLDivElement>(null);
  const carouselMask = useRef<HTMLDivElement>(null);
 
  const animationFrameIdRef = useRef<number | null>(null);
  const animationPrevTimeRef = useRef<number | null>(null);
  const cardElementRefs = useRef<CardElementsRefs>([]);
 
  const expressBaseUrl = getExpressNewProjectUrl();
 
  const textEffectsUrl = isMobileOrTablet
    ? getExpressTextEffectsUrlMobile()
    : `${expressBaseUrl}?${intl.formatMessage(
        moduleMessages.textEffectsUrlParams,
      )}`;
 
  const templateUrl = isMobileOrTablet
    ? getExpressTemplateGenerateUrlMobile()
    : `${expressBaseUrl}?${intl.formatMessage(
        moduleMessages.textTemplateUrlParams,
      )}`;
 
  const EXTERNAL_TOOLS: Array<CandidateInfo> = [
    {
      image: templateImage,
      video: templateVideo,
      title: intl.formatMessage(moduleMessages.templateTitle),
      description: intl.formatMessage(moduleMessages.templateDescription),
      id: ModuleId.Text2Template,
      accessIcon: ExpressIcon,
      accessIconAlt: "Adobe Express",
      cta: {
        label: intl.formatMessage(moduleMessages.tryInExpress),
        url: templateUrl,
      },
      showOnMobile: true,
      useHref: isMobileOrTablet,
    },
    {
      image: vectorCreationImage,
      title: intl.formatMessage(moduleMessages.textVectorTitle),
      description: intl.formatMessage(moduleMessages.textVectorDescription),
      accessIcon: IllustratorIcon,
      accessIconAlt: "Adobe Illustrator",
      cta: {
        label: intl.formatMessage(moduleMessages.learnMore),
        url: intl.formatMessage(moduleMessages.textVectorGoUrl),
      },
      id: ModuleId.Text2Vector,
      showOnMobile: true,
    },
    {
      image: recolorImage,
      video: recolorVideo,
      title: intl.formatMessage(moduleMessages.recolorTitle),
      description: intl.formatMessage(moduleMessages.recolorDescription),
      accessIcon: IllustratorIcon,
      accessIconAlt: "Adobe Illustrator",
      cta: {
        label: intl.formatMessage(moduleMessages.tryInIllustrator),
        url: isMobileOrTablet
          ? intl.formatMessage(moduleMessages.recolorGoUrl)
          : intl.formatMessage(moduleMessages.recolorDeepLink),
      },
      id: ModuleId.Recolor,
      showOnMobile: true,
    },
    {
      image: textEffectsImage,
      video: textEffectsVideo,
      title: intl.formatMessage(moduleMessages.textEffectsTitle),
      description: intl.formatMessage(moduleMessages.textEffectsDescription),
      accessIcon: ExpressIcon,
      accessIconAlt: "Adobe Express",
      cta: {
        label: intl.formatMessage(moduleMessages.tryInExpress),
        url: textEffectsUrl,
      },
      id: ModuleId.TextEffects,
      showOnMobile: true,
      useHref: isMobileOrTablet,
    },
  ];
 
  function checkActiveModuleOnMobile() {
    let moduleAtCenter = "";
    let moduleCenterDistance = 100000;
 
    Object.values(cardElementRefs.current).forEach(
      ({ id, cardRef }: CardElementRef) => {
        if (cardRef.current) {
          const rect = cardRef.current.getBoundingClientRect();
          if (
            rect.top >= 0 &&
            rect.left >= 0 &&
            rect.bottom <= window.innerHeight &&
            rect.right <= window.innerWidth
          ) {
            const distance = Math.abs(
              rect.x + rect.width / 2 - window.innerWidth / 2,
            );
            if (moduleCenterDistance > distance) {
              moduleAtCenter = id;
              moduleCenterDistance = distance;
            }
          }
        }
      },
    );
    setActiveModuleId(moduleAtCenter);
  }
 
  // Step of the carousel scroll animation
  const scrollStep = (time: number | undefined) => {
    if (!animationPrevTimeRef.current) {
      animationPrevTimeRef.current = time ?? 0;
    }
    if (time && carouselMask.current) {
      const delta = time - animationPrevTimeRef.current;
      const newScrollPosition = carouselMask.current.scrollLeft + delta * 0.12;
      carouselMask.current.scrollLeft = newScrollPosition;
      animationPrevTimeRef.current = time;
    }
    animationFrameIdRef.current = requestAnimationFrame(scrollStep);
  };
 
  // Start automatic scrolling
  const startScrolling = () => {
    if (!isMobile && isAutoScrollEnabled) {
      animationFrameIdRef.current = requestAnimationFrame(scrollStep);
    } else if (animationFrameIdRef.current) {
      cancelAnimationFrame(animationFrameIdRef.current);
    }
  };
 
  const stopScrolling = () => {
    // Cancel the ongoing animation frame to pause the animation
    if (animationFrameIdRef.current) {
      cancelAnimationFrame(animationFrameIdRef.current);
    }
    // Reset the prev time to avoid a jump in the animation
    if (animationPrevTimeRef.current) {
      animationPrevTimeRef.current = null;
    }
  };
 
  useEffect(() => {
    if (isMobile && carouselInner.current) {
      const callback = (_headings: Array<IntersectionObserverEntry>) => {
        checkActiveModuleOnMobile();
      };
 
      const observer = new IntersectionObserver(callback, { threshold: 1 });
      observer.observe(carouselInner.current);
      return () => observer.disconnect();
    }
  }, [isMobile, carouselInner.current]);
 
  useEffect(() => {
    // Start continuous scrolling when the component mounts
    startScrolling();
 
    // Cleanup on component unmount
    return () => stopScrolling();
  }, [isMobile, isAutoScrollEnabled]);
 
  const activeCards = useMemo<Array<ModuleCardInfo>>(() => {
    const isVisibleByConfig = (module: ModuleConfig) => {
      const isFloodgateFlagEnabled = (flag: FloodgateFlagName) =>
        isFeatureEnabled(flag);
      if (module?.checkVisibilityOfCard) {
        return module.checkVisibilityOfCard(
          isFloodgateFlagEnabled,
          isFulfillableItemEnabled,
        );
      }
      return true;
    };
 
    const activeModules = registry.registeredModules.filter((module) => {
      const moduleStatus = detectModuleStatus(module);
      const { isVisible, isRestricted } = {
        [ModuleNavigationStatus.Undetermined]: {},
        [ModuleNavigationStatus.Hidden]: {},
        [ModuleNavigationStatus.Retired]: {},
        [ModuleNavigationStatus.Restricted]: {
          isVisible: !isMobile, // Hide restricted modules on mobile.
          isRestricted: true,
        },
        [ModuleNavigationStatus.Available]: { isVisible: true },
        [ModuleNavigationStatus.RestrictFeature]: { isVisible: true },
      }[moduleStatus];
 
      return isVisible && !isRestricted && isVisibleByConfig(module);
    });
 
    const activeExternalTools = EXTERNAL_TOOLS.filter((tool) => {
      return !tool.disabled && (!isMobile || tool.showOnMobile) && tool.cta;
    });
 
    return [
      ...activeModules.map((module) => {
        return {
          title: intl.formatMessage(module.title),
          id: module.id,
          description: intl.formatMessage(module.description),
          image: module.image,
          video: module.video,
          accessIcon: module.accessIcon,
          accessIconAlt: module.accessIconAlt,
          ctaLabel: module.ctaLabel
            ? intl.formatMessage(module.ctaLabel)
            : undefined,
          ctaUrl: module.defaultModuleRoute,
          showAccessLabel: module.showAccessLabel ?? false,
          accessLabel: module.accessLabel
            ? intl.formatMessage(module.accessLabel)
            : undefined,
        };
      }),
      ...activeExternalTools.map((tool) => {
        return {
          title: tool.title,
          id: tool.id,
          description: tool.description,
          image: tool.image,
          video: tool.video,
          accessIcon: tool.accessIcon,
          accessIconAlt: tool.accessIconAlt,
          ctaLabel: tool.cta?.label,
          ctaUrl: tool.cta?.url,
          useHref: tool.useHref,
        };
      }),
    ];
  }, [
    isMobile,
    registry.registeredModules,
    floodgateFlagsWithOverrides,
    isFulfillableItemEnabled,
  ]);
 
  useEffect(() => {
    if (
      infinite &&
      carouselInner.current &&
      carouselMask.current &&
      activeCards.length > 0
    ) {
      const singleItemWidth = (
        carouselInner.current.lastElementChild as HTMLElement
      ).offsetWidth;
      const initialScrollPosition = singleItemWidth * CLONED_ITEM_COUNT;
      carouselMask.current.scrollTo({
        left: initialScrollPosition,
        behavior: "auto",
      });
    }
  }, [activeCards.length, infinite]);
 
  const handleScroll = () => {
    if (infinite && carouselInner.current && carouselMask.current) {
      const scrollX = carouselMask.current.scrollLeft;
      const lastChild = carouselInner.current.lastElementChild as HTMLElement;
      const itemWidth = lastChild.offsetWidth + GAP;
      // Calculate when to jump from the end clones to the start of the original list
      if (
        scrollX >=
        itemWidth * CLONED_ITEM_COUNT + itemWidth * activeCards.length
      ) {
        const newPosition = itemWidth * CLONED_ITEM_COUNT;
        carouselMask.current.scrollLeft = newPosition;
      }
      // Calculate when to jump from the start clones to the end of the original list
      else if (scrollX <= itemWidth * (CLONED_ITEM_COUNT - 1)) {
        const newPosition =
          itemWidth * CLONED_ITEM_COUNT + itemWidth * (activeCards.length - 1);
        carouselMask.current.scrollLeft = newPosition;
      }
    }
 
    if (isMobile) {
      checkActiveModuleOnMobile();
    }
  };
 
  // Copy some of the items to the front and back to allow infinite scrolling
  const clonedFrontItems = () => {
    return expandToLength(activeCards, CLONED_ITEM_COUNT, false);
  };
  const clonedBackItems = () => {
    return expandToLength(activeCards, CLONED_ITEM_COUNT, true);
  };
  const realItems = () =>
    activeCards.map((card) => ({
      ...card,
      dataTestId: `${card.id}-card`,
    }));
 
  const cardsList: Array<ModuleCardInfo> = infinite
    ? [...clonedFrontItems(), ...realItems(), ...clonedBackItems()]
    : realItems();
 
  const cmeEnabled = hasCustomModelAccess(
    isFulfillableItemEnabled,
    floodgateFlagsWithOverrides,
  );
 
  return (
    <ModuleCarousel
      moduleList={cardsList}
      heading={intl.formatMessage(moduleMessages.carouselHeading)}
      description={intl.formatMessage(moduleMessages.carouselDescription)}
      activeModuleId={activeModuleId}
      isMobile={isMobile}
      cardElementRefs={cardElementRefs}
      carouselInner={carouselInner}
      carouselMask={carouselMask}
      cmeEnabled={cmeEnabled}
      scrollCallback={handleScroll}
      startScrollAnimation={startScrolling}
      stopScrollAnimation={stopScrolling}
      nextPageAriaLabel={intl.formatMessage(moduleMessages.nextModulePage)}
      prevPageAriaLabel={intl.formatMessage(moduleMessages.prevModulePage)}
