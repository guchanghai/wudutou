const componentMap = {

    formattedLabel,

    validationPattern?: string;
  }) => {

    const [currentValue, setCurrentValue] = useState<string>(value as string);
    return (
      <>

        <Textfield
          aria-label={formattedLabel}
          className={clsx(styles.feedbackElement, styles.feedbackTextField)}
          value={currentValue}
          input={(event) => {
            if (!textFieldRequired && required) {
              // Do not initially flag the field as required until it has received input
              setTextFieldRequired(true);
            }
            setCurrentValue((event.currentTarget as TextfieldType).value);
            onChange((event.currentTarget as TextfieldType).value);
          }}
          required={textFieldRequired}
          type={textFieldValidationType}
          pattern={validationPattern}
        />
      </>
    );
  },

const getInitialValueByType = (
  type: FeedbackQuestionsType,
): FeedbackChangeValue => {
  switch (type) {
    case FeedbackQuestionsType.Text:
    case FeedbackQuestionsType.TextArea:
    case FeedbackQuestionsType.SingleSelect:
    case FeedbackQuestionsType.Title:
    case FeedbackQuestionsType.TitleWithLink:
      return "";
    case FeedbackQuestionsType.MultiSelect:
      return [];
    case FeedbackQuestionsType.MultiSelectCollapse:
    case FeedbackQuestionsType.Checkbox:
      return false;
    case FeedbackQuestionsType.ImageCheckbox:
      return true;
  }
};
 
const getInitialValueForPrefill = (
  question: FeedbackQuestion,
  userProfile?: UserProfile | null,
) => {
  if (
    question.type === FeedbackQuestionsType.Text &&
    question.id === FeedbackQuestionId.ReportAbuseEmail
  ) {
    return userProfile?.email || "";
  }
  if (
    question.type === FeedbackQuestionsType.Text &&
    question.id === FeedbackQuestionId.ReportAbuseName
  ) {
    return userProfile?.name || "";
  }
  return getInitialValueByType(question.type);
};

type FeedbackModalState = Record<
  string,
  | {
      value: FeedbackChangeValue;
      type: FeedbackQuestionsType;
      isConfigQuestion: boolean;
      required: boolean;
    }
  | undefined
>;
 
export function FeedbackModal(props: Props) {
  const intl = useIntl();
  const { userProfile } = useAuthContext();
 
  const form = feedbackRegistryMap[props.feedbackId][props.feedbackType];
 
  const {
    feedback: {
      internal: { feedbackForm },
    },
  } = useFeedbackContext();
 
  const questionState = useRef<FeedbackModalState>({});
  const [optionsState, setOptionState] = useState<
    Record<string, Array<string>>
  >({});
  const [isSubmitBtnDisabled, setIsSubmitButtonDisabled] =
    useState<boolean>(true);
 
  // Escape key handler to prevent the rest of the app from handling ESC
  // when this modal is active.
  const handleEscape = (event: KeyboardEvent) => {
    if (event.key === "Escape") {
      event.stopPropagation();
    }
  };
 
  useEffect(() => {
    document.addEventListener("keydown", handleEscape, false);
    return () => {
      document.removeEventListener("keydown", handleEscape, false);
    };
  }, []);
 
  const handleChange = (
    question: FeedbackQuestion,
    value: FeedbackChangeValue,
  ) => {
    questionState.current = {
      ...questionState.current,
      [question.id]: {
        value,
        type: question.type,
        isConfigQuestion: question.isConfigQuestion ?? false,
        required: question.required,
      },
    };
    setIsSubmitButtonDisabled(!validateResponse());
  };
 
  const handleChangeOptions = (
    question: FeedbackQuestion,
    value: Array<string>,
  ) => {
    // This is a special case for the MultiSelectCollapse component, when there's no checkbox at the top
    // and only options are available, we need to update the state of the question
    if (
      question.type === FeedbackQuestionsType.MultiSelectCollapse &&
      question.hideLabel
    ) {
      const newValue = value.length > 0;
      handleChange(question, newValue);
    }
 
    // Update options state
    setOptionState((prev) => ({ ...prev, [question.id]: value }));
  };
 
  function getStateProperty(question: FeedbackQuestion): FeedbackChangeValue {
    const item = questionState.current[question.id];
    const value =
      item?.value ||
      getInitialValueByType(question.type) ||
      getInitialValueForPrefill(question, userProfile);
    if (!item) {
      questionState.current = {
        ...questionState.current,
        [question.id]: {
          value,
          type: question.type,
          isConfigQuestion: question.isConfigQuestion ?? false,
          required: question.required,
        },
      };
    }
    return value;
  }
 
  function formatQuestionOptions(
    question: FeedbackQuestion,
  ): Array<{ label: string; value: string }> {
    if (
      question.type === FeedbackQuestionsType.MultiSelectCollapse ||
      question.type === FeedbackQuestionsType.MultiSelect ||
      question.type === FeedbackQuestionsType.SingleSelect
    ) {
      if (question.options) {
        return question.options.map((item) => ({
          label: intl.formatMessage(item.label),
          value: item.value,
        }));
      }
    }
 
    return [];
  }
 
  /**
   * In some cases we do not want the displayed option string to be the value we submit to a service or other API.
   * We need to associate the formatted (translated) display string to a stable value for submission.
   * @param question
   * @param optionValuesMap A map of react-intl MessageDescriptor keys to stable non-translated values
   * e.g. "@clio/playground:shared-link-feed-back-form-report-option-2": "hate"
   * @returns Map<string, string> mapping translated string to the stable option value
   * e.g. "Racist or Hate Content": "hate"
   */
  function buildQuestionOptionValues(
    question: FeedbackQuestion,
    optionValuesMap?: Map<string, string>,
  ) {
    const optionValues = new Map();
    if (
      question.type === FeedbackQuestionsType.MultiSelectCollapse ||
      question.type === FeedbackQuestionsType.MultiSelect ||
      question.type === FeedbackQuestionsType.SingleSelect
    ) {
      if (question.options && optionValuesMap) {
        question.options.forEach((item) => {
          if (item.label.id) {
            optionValues.set(
              intl.formatMessage(item.label),
              optionValuesMap.get(item.label.id),
            );
          }
        });
      }
    }
 
    return optionValues;
  }
 
  function getComponent<K extends keyof typeof componentMap>(key: K) {
    return componentMap[key];
  }
 
  /**
   * Validate if at least one question is answered
   * @returns true if at least one question is answered
   */
  const validateResponse = () => {
    const questionKeys = Object.keys(questionState.current);
    if (!questionKeys.length) {
      return false;
    }
 
    const questionValues = Object.values(questionState.current);
 
    const hasUnansweredRequired = questionValues.some((item) => {
      if (
        item?.required &&
        (!item.value || (Array.isArray(item.value) && item.value.length === 0))
      ) {
        return true;
      }
      return false;
    });
 
    return (
      !hasUnansweredRequired &&
      questionValues.some((item) => {
        // Stop if the value is undefined
        // Stop if the value is an empty array
        if (
          item?.value === undefined ||
          (Array.isArray(item.value) && item.value.length === 0)
        ) {
          return false;
        }
 
        if (
          item.type === FeedbackQuestionsType.TextArea &&
          questionKeys.length === 1
        ) {
          return false;
        }
 
        return true;
      })
    );
  };
 
  /**
   * Create an array of objects with questionId, value and options
   * @returns array of objects with questionId, value and options
   */
  function buildResponseData() {
    return Object.entries(questionState.current)
      .map(([key, item]) => {
        // This should never happen, but just in case, validateResponse should take care of it
        if (item?.value === undefined || item.value === "") {
          return null;
        }
 
        return {
          questionId: key,
          value: item.value,
          options: optionsState[key],
          type: item.type,
          isConfigQuestion: item.isConfigQuestion,
        };
      })
      .filter(Boolean) as Array<FeedbackResponse>;
  }
 
  /**
   * Submit the form and call the onSubmit function
   */
  function handleSubmit() {
    if (!validateResponse()) {
      setIsSubmitButtonDisabled(true);
      return;
    }
 
    const res = buildResponseData();
    void props.onSubmit(res);
    if (props.feedbackType === "report") {
      props.onReportSuccess?.();
    }
  }
 
  return (
    <OverlayTrigger
      type="modal"
      open="click"
      spClosed={() => {
        focusGalleryModal();
        props.onClose();
      }}
    >
      <span slot="trigger"></span>
 
      <DialogBase
        tabIndex={0}
        slot="click-content"
        underlay
        onClose={() => {
          focusGalleryModal();
          props.onClose();
        }}
      >
        <Dialog
          className={styles.feedbackDialog}
          id={FEEDBACK_MODEL_ID}
          size="m"
          noDivider
          data-testid="feedback-modal"
        >
          <Heading size="s" slot="heading" className={styles.feedbackHeading}>
            {intl.formatMessage(form.label)}
          </Heading>
 
          <form
            onSubmit={(e) => {
              e.preventDefault();
              handleSubmit();
            }}
            className={styles.feedbackForm}
            data-testid="feedback-modal-form"
          >
            <Grid columns="1fr" gap="size-100">
              {form.questions.map((question: FeedbackQuestion) => {
                const Component = getComponent(question.type);
 
                const value = getStateProperty(question);
 
                const hasOptionsProps =
                  question.type === FeedbackQuestionsType.MultiSelectCollapse;
                const formattedOptions = formatQuestionOptions(question);
                const optionValueMap =
                  question.type === FeedbackQuestionsType.SingleSelect
                    ? buildQuestionOptionValues(
                        question,
                        question.optionValuesMap,
                      )
                    : new Map();
 
                const formattedLabel = intl.formatMessage(question.label);
                const placeholder = intl.formatMessage({
                  id: "@clio/playground:feedback-modal-add-note",
                  defaultMessage: "Add a note (Optional)",
                });
                return (
                  <>
                    <Component
                      key={question.id}
                      {...question}
                      formattedLabel={formattedLabel}
                      placeholder={placeholder}
                      formattedOptions={formattedOptions}
                      optionValuesMap={optionValueMap}
                      onChange={(newValue: FeedbackChangeValue) =>
                        handleChange(question, newValue)
                      }
                      optionsKeys={
                        hasOptionsProps ? optionsState[question.id] : undefined
                      }
                      onChangeOptions={(newValue) => {
                        if (hasOptionsProps) {
                          handleChangeOptions(question, newValue);
                        }
                      }}
                      value={value}
                      resultImage={feedbackForm?.result}
                      referenceImage={feedbackForm?.referenceImage}
                      question={question}
                      getStateProperty={getStateProperty}
                      feedbackType={props.feedbackType}
                    />
                    {question.divider && (
                      <Divider size="m" key={`${question.id}-divider`} />
                    )}
                  </>
                );
              })}
            </Grid>
          </form>
 
          <Flex slot="button" gap="size-100">
            <SWCModalButton
              variant="secondary"
              treatment="outline"
              onClick={() => props.onClose()}
              autoFocus={false}
            >
              {intl.formatMessage({
                id: "@clio/playground:feedback-modal-cancel",
                defaultMessage: "Cancel",
              })}
            </SWCModalButton>
            <SWCModalButton
              disabled={isSubmitBtnDisabled}
              variant="accent"
              onClick={() => handleSubmit()}
            >
              {props.feedbackType === "report"
                ? intl.formatMessage({
                    id: "@clio/playground:feedback-modal-submit-report",
                    defaultMessage: "Submit report",
                  })
                : intl.formatMessage({
                    id: "@clio/playground:feedback-modal-submit-feedback",
                    defaultMessage: "Submit feedback",
                  })}
            </SWCModalButton>
          </Flex>
        </Dialog>
      </DialogBase>
    </OverlayTrigger>
  );
}
