type UDSDocument = {
  data: Record<string, unknown>;
  version: number;
};
 
/**
 * options required to access Uds
 * appDomain - Uds app domain
 * docType - Uds docType, by default it is '$' at Uds end
 */
export type UDSOptions = {
  appDomain: string;
  docType?: string;
};
 
export class UDSClient {
  private _accessToken = "";
  private _userId = "";
  private _isSignedInUser = false;
  private _initialized = false;
 
  private _fetchDocumentPromiseMap: Map<string, Promise<UDSDocument>>;
  private _documentCache: Map<string, UDSDocument | undefined>;
 
  constructor() {
    this._fetchDocumentPromiseMap = new Map();
    this._documentCache = new Map();
  }
 
  // user document is unique for appDomain and docType, default docType = '$' at UDS end
  private _getDocumentUID(appDomain: string, docType?: string) {
    if (docType) {
      return appDomain + ":" + docType;
    }
    return appDomain;
  }
 
  private _getAppDomain(options: UDSOptions) {
    const appDomain = options.appDomain;
    if (!appDomain.length) {
      throw new Error("Invalid appDomain");
    }
    return appDomain;
  }
 
  private _getCachedDocument(cacheKey: string): UDSDocument | undefined {
    return this._documentCache.get(cacheKey);
  }
 
  private _setCachedDocument(
    cacheKey: string,
    document: UDSDocument | undefined,
  ) {
    if (document) {
      this._documentCache.set(cacheKey, document);
    }
  }
 
  private async _fetchDocumentUtil(
    accessToken: string,
    options: UDSOptions,
    forceFetch = false,
  ): Promise<UDSDocument | undefined> {
    const appDomain = this._getAppDomain(options);
    const docType = options.docType;
    const documentId = this._getDocumentUID(appDomain, docType);
 
    let document = this._getCachedDocument(documentId);
 
    if (forceFetch || !document) {
      let fetchDocumentPromise = this._fetchDocumentPromiseMap.get(documentId);
      if (!fetchDocumentPromise) {
        fetchDocumentPromise = this._fetchDocument(accessToken, options);
        this._fetchDocumentPromiseMap.set(documentId, fetchDocumentPromise);
      }
      document = await fetchDocumentPromise;
    }
    return document;
  }
 
  private _addQueryParam(url: string, key: string, value: string): string {
    if (url && key && value) {
      const urlObj = new URL(url);
      urlObj.searchParams.set(key, value);
      return urlObj.href;
    }
    return url;
  }
 
  private _getUrl(
    appDomain: string,
    docType?: string,
    version?: number,
  ): string {
    let url = `${UDS_HOST}/userdocs/${appDomain}`;
    if (docType) {
      url = this._addQueryParam(url, "docType", docType);
    }
    if (version !== undefined && version >= 0) {
      url = this._addQueryParam(url, "version", version.toString());
    }
    return url;
  }
 
  private async _fetchDocument(
    accessToken: string,
    options: UDSOptions,
  ): Promise<UDSDocument> {
    const appDomain = options.appDomain;
    const docType = options.docType;
 
    const fetchRequestOptions: RequestInit = {
      method: "GET",
      headers: new Headers({
        Accept: "*/*",
        "x-api-key": IMSConfig.clientId,
        Authorization: `Bearer ${accessToken}`,
        "Access-Control-Allow-Origin": window.location.origin,
      }),
      mode: "cors",
    };
 
    let document;
    try {
      const url = this._getUrl(appDomain, docType);
      const response = await fetchWithRetry(url, fetchRequestOptions);
      if (response.status !== 200) {
        throw response;
      }
 
      document = await response.json();
    } catch (error) {
      if (error instanceof Response && error.status === 404) {
        // UDS API returns 404 if document does not exist, return empty document in this case with version 0 (virtual doc exist in cache only)
        document = {
          version: 0,
          data: {},
        };
      } else {
        throw error;
      }
    }
 
    const documentId = this._getDocumentUID(appDomain, docType);
    this._setCachedDocument(documentId, document);
    this._fetchDocumentPromiseMap.delete(documentId);
    return document;
  }
 
  private _isSafeToSave(
    sectionName: string,
    newDocument: UDSDocument,
    cachedDocument?: UDSDocument,
  ): boolean {
    if (!cachedDocument) {
      return true;
    }
    const cachedSectionData = cachedDocument.data[sectionName];
    const newSectionData = newDocument.data[sectionName];
    const cachedSectionDataType = typeof cachedSectionData;
    const newSectionDataType = typeof newSectionData;
    if (cachedSectionDataType !== newSectionDataType) {
      return false;
    }
    if (cachedSectionDataType === "object") {
      return isEqual(cachedSectionData, newSectionData);
    }
    return cachedSectionData === newSectionData;
  }
 
  /**
   * Initializes the UDS client
   * @param accessToken - Access token of the user
   * @param isSignedIn - If true, user is signed in
   * @param userId - User id
   */
  public init(
    accessToken: string,
    isSignedIn: boolean,
    userProfile?: UserProfile,
  ) {
    this._accessToken = accessToken;
    this._isSignedInUser = isSignedIn;
    this._userId = userProfile?.userId ?? "";
    this._initialized =
      accessToken.length > 0 && isSignedIn && userProfile !== undefined;
  }
 
  /**
   * Returns true if the UDS client is initialized
   */
  public get isInitialized(): boolean {
    return this._initialized;
  }
 
  /**
   * Fetches the UDS data for the given section name
   * @param sectionName - Name of the section
   * @param forceFetch - If true, fetches the data from UDS server
   * @param options - Options required to access UDS
   * @returns Promise of the section data
   * @throws Error if the section data is not found
   */
  public async fetchAppSectionData<SectionDataType>(
    sectionName: string,
    forceFetch = false,
    options?: UDSOptions,
  ): Promise<SectionDataType | undefined> {
    if (!this._isSignedInUser) {
      console.debug("Unable to fetch UDS data for anonymous user");
      return undefined;
    }
    options = options ?? { appDomain: UDS_APP_DOMAIN };
    const document = await this._fetchDocumentUtil(
      this._accessToken,
      options,
      forceFetch,
    );
    if (document) {
      return document.data[sectionName] as SectionDataType | undefined;
    }
    return undefined;
  }
 
  /**
   * Saves the UDS data for the given section name
   * @param sectionName - Name of the section
   * @param sectionData - Data to be saved
   * @param forceFetch - If true, fetches the data from UDS server
   * @param options - Options required to access UDS
   * @param cachedDocument - Cached document
   * @returns Promise of the section data
   * @throws Error if the user is not signed in or if the section data is not found
   */
  public async saveAppSectionData<SectionDataType>(
    sectionName: string,
    sectionData: SectionDataType,
    forceFetch = false,
    options?: UDSOptions,
    cachedDocument?: UDSDocument,
  ): Promise<Record<string, unknown> | undefined> {
    if (!this._isSignedInUser) {
      throw new Error("Unable to save UDS data for anonymous user");
    }
 
    options = options ?? { appDomain: UDS_APP_DOMAIN };
    const appDomain = this._getAppDomain(options);
    const docType = options.docType;
 
    const documentData = {
      [sectionName]: sectionData,
    };
 
    const createDocumentRequestBody = {
      documentId: {
        appDomain: `${appDomain}`,
        ownerEntity: `${this._userId}`,
        docType,
      },
      data: documentData,
    };
 
    let document;
    try {
      document = await this._fetchDocumentUtil(
        this._accessToken,
        options,
        forceFetch,
      );
      if (document) {
        const isSafeToSave = this._isSafeToSave(
          sectionName,
          document,
          cachedDocument,
        );
        if (!isSafeToSave) {
          throw new Error(
            "Unable to save section data due to concurrent write",
          );
        }
 
        const version = document.version;
        const url = this._getUrl(appDomain, docType, version);
 
        const fetchRequestOptions: RequestInit = {
          method: "PATCH",
          body: JSON.stringify(createDocumentRequestBody),
          headers: new Headers({
            Accept: "*/*",
            "x-api-key": IMSConfig.clientId,
            Authorization: `Bearer ${this._accessToken}`,
            "Content-Type": "application/json",
            "Access-Control-Allow-Origin": window.location.origin,
          }),
          mode: "cors",
        };
 
        const response = await fetchWithRetry(url, fetchRequestOptions);
        if (response.status !== 200 && response.status !== 201) {
          throw response;
        }
        // delete cache
        const documentId = this._getDocumentUID(appDomain, docType);
        this._documentCache.delete(documentId);
        return response.json();
      }
    } catch (error) {
      // in case of conflict => read fresh document and update on latest version
      if (error instanceof Response && error.status === 409) {
        return await this.saveAppSectionData(
          sectionName,
          sectionData,
          true,
          options,
          document,
        );
      }
      throw error;
    }
  }
 
  public async deleteDocument(options?: UDSOptions) {
    if (!this._isSignedInUser) {
      throw new Error("Unable to save UDS data for anonymous user");
    }
 
    options = options ?? { appDomain: UDS_APP_DOMAIN };
    const appDomain = this._getAppDomain(options);
    const docType = options.docType;
    const url = this._getUrl(appDomain, docType);
 
    const fetchRequestOptions: RequestInit = {
      method: "DELETE",
      headers: new Headers({
        Accept: "*/*",
        "x-api-key": IMSConfig.clientId,
        Authorization: `Bearer ${this._accessToken}`,
        "Access-Control-Allow-Origin": window.location.origin,
      }),
      mode: "cors",
    };
 
    const response = await fetchWithRetry(url, fetchRequestOptions);
    if (response.status !== 200 && response.status !== 204) {
      throw new Error(
        `UDS deleting failure, ${response.status} ${response.statusText}`,
      );
    }
 
    // delete cache
    const documentId = this._getDocumentUID(appDomain, docType);
    this._documentCache.delete(documentId);
  }
