/**
 * Available locks that can be used throughout the app to orchestrate async behavior. Add new locks here.
 */
export enum Lock {
  SYNC_OPERATION = "SYNC_OPERATION",
}
 
/**
 * Lock modes. Define the behavior of locked functions when wrapping them.
 * Bitwise values so you can mix and match.
 */
export enum LockMode {
  // Functions locked with this mode will only check the lock before executing, but will not enforce the lock during execution.
  // This is the recommended mode for wrapping low-priority functions you want to prevent from executing too early.
  WAIT_ONLY = 0,
  // Functions locked with this mode will both check for the lock before executing and enforce the lock during execution.
  // This is the recommended mode for wrapping high-priority functions you want to wait on.
  // Will override WAIT_ONLY.
  // Note: This WON'T pause execution of other functions that were locked with WAIT_ONLY and were started prior to this. Those functions will continue their execution unguarded!
  ENFORCE_LOCK = 1 << 0,
  // Drop pending calls while lock is active.
  // If no call was made during the lock, nothing will happen. If multiple calls were made, we only run the last one after the lock is lifted.
  DROP_PENDING_CALLS = 1 << 1,
  // If you only care about the last call, specify this flag and a supplied cancel function that can abort the initial call and free the lock faster.
  // We will call this function only once per guarded run: when the lock is first tried and found busy. It may once again be called once the lock is released.
  // The cancel function must be synchronous.
  USE_CANCEL_FUNCTION = 1 << 2,
}
 
/**
 * No-op lock. This can be used in place of check, acquire or release so that it doesn't perform any operation.
 *
 * @returns Returns false so that when used as a check, the lock will be seen as open.
 */
export function noopLock(): boolean {
  return false;
}

/**
 * Lock a function using a specific lock from the Locks enum, as specified by the LockMode option.
 * The function will check the lock and reschedule itself if the lock is set, retrying after the
 * specified interval (ms). If the lock mode is set to enforce the lock, then it will set the lock upon
 * beginning execution and release it after finishing; if the lock mode is set to wait only, it will
 * leave the lock unset.
 *
 * @param func Function to wrap with the lock check, acquire and release functions.
 * @param lock The name of the lock to use.
 * @param lockMode The lock mode to use. This specifies whether the lock should only be read or also be set during execution.
 * @param checkInterval In case the lock is in use when trying to execute, this specifies how long (in ms) to wait before checking again. Defaults to 0.
 * @param cancelFn A (synchronous!) function that will cancel the current run to accelerate the release of the lock. This is guaranteed to only be called once. Can be left undefined.
 * @param onStatusChanged A callback which signals the status of the lock. This can be used to show an indicator that is present for the duration of the entire guarded operation. Will be called with 'true' when the operation starts (i.e. lock becomes busy for the first time) and 'false' when it stops (i.e. lock becomes free for the last time).
 * @returns The function wrapped in lock attire.
 */
export function lockFunction(
  func: (...args: Array<unknown>) => unknown,
  lock: Lock,
  lockMode: LockMode,
  checkInterval = 0,
  cancelFn?: () => void,
  onStatusChanged?: (value: boolean, isLast: boolean) => void,
): (...args: Array<unknown>) => unknown {
  const acquireLockFn =
    lockMode & LockMode.WAIT_ONLY
      ? noopLock
      : acquireLock(lock, onStatusChanged);
  const releaseLockFn =
    lockMode & LockMode.WAIT_ONLY
      ? noopLock
      : releaseLock(lock, onStatusChanged);
  const dropPendingCalls = !!(lockMode & LockMode.DROP_PENDING_CALLS);
 
  return lockFunctionRaw(
    func,
    checkLock(lock),
    acquireLockFn,
    releaseLockFn,
    checkInterval,
    dropPendingCalls,
    lockMode & LockMode.USE_CANCEL_FUNCTION ? cancelFn : undefined,
  );
}
