import { jsonToHash } from "shared/data/hashing";
import { assert } from "shared/utils/ClioError";
import { Deferred } from "shared/utils/Deferred";
 
export type RequestWithSeed = {
  seeds?: Array<number>;
  n?: number;
};
 
export type ResponseImage = {
  seed: number;
  /* eslint-disable-next-line @typescript-eslint/no-explicit-any -- image structure is defined by caller */
  image: any;
};
 
export type ResponseWithOutputs = {
  outputs: Array<ResponseImage>;
};
 
export type ExecuteJobFn<
  RequestT extends RequestWithSeed,
  ResponseT extends ResponseWithOutputs,
> = (request: RequestT, signal: AbortSignal) => Promise<ResponseT>;
 
type Job<
  RequestT extends RequestWithSeed,
  ResponseT extends ResponseWithOutputs,
> = {
  request: RequestT;
  seed: number;
  signal: AbortSignal;
  deferred: Deferred<ResponseT>;
};
 
async function computeHashForModel<RequestT extends RequestWithSeed>(
  input: RequestT,
) {
  return await jsonToHash(input);
}
 
export class FireflyBatchManager<
  RequestT extends RequestWithSeed,
  ResponseT extends ResponseWithOutputs,
> {
  private _timeoutIds = new Map<string, ReturnType<typeof setTimeout>>();
  private _queues = new Map<string, Array<Job<RequestT, ResponseT>>>();
  constructor(
    private maxSeeds: number,
    private executeFn: ExecuteJobFn<RequestT, ResponseT>,
  ) {}
 
  async executeWithBatch(
    request: RequestT,
    seed: number,
    signal: AbortSignal,
  ): Promise<ResponseT> {
    assert(
      !request.seeds && request.n === undefined,
      "Request already has seeds or n",
    );
    const deferred = new Deferred<ResponseT>();
    const key = await computeHashForModel(request);
    this._addRequestToQueue(key, { request, seed, signal, deferred });
    this._scheduleNextBatchIfNeeded(key);
    return await deferred.promise;
  }
 
  private _addRequestToQueue(key: string, job: Job<RequestT, ResponseT>) {
    if (!this._queues.has(key)) {
      this._queues.set(key, []);
    }
    this._queues.get(key)!.push(job);
  }
 
  private async _executeBatch(jobs: Array<Job<RequestT, ResponseT>>) {
    assert(jobs.length <= this.maxSeeds && jobs.length > 0, "Too many items");
    const controller = new AbortController();
    let activeJobs = jobs.length;
    for (const job of jobs) {
      assert(!job.signal.aborted, "Job is not supposed to be aborted");
      const abortHandler = () => {
        job.signal.removeEventListener("abort", abortHandler);
        this._abortJob(job);
        activeJobs--;
        if (activeJobs === 0) {
          controller.abort();
        }
      };
      job.signal.addEventListener("abort", abortHandler);
    }
    try {
      const batchRequest = {
        ...jobs[0].request,
        seeds: jobs.map((job) => job.seed),
      };
      const result = await this.executeFn(batchRequest, controller.signal);
      for (const job of jobs) {
        if (!job.signal.aborted) {
          job.deferred.resolve(result);
        }
      }
    } catch (e) {
      for (const job of jobs) {
        if (!job.signal.aborted) {
          job.deferred.reject(e);
        }
      }
    }
  }
 
  private _abortJob(job: Job<RequestT, ResponseT>) {
    job.deferred.reject(
      new DOMException("The request aborted by user", "AbortError"),
    );
  }
 
  private _scheduleBatchWork(key: string) {
    const queue = this._queues.get(key);
    while (queue && queue.length > 0) {
      const queueItems = queue.splice(0, this.maxSeeds);
      const itemsToRun = queueItems.filter((item) => {
        if (item.signal.aborted) {
          this._abortJob(item);
          return false;
        }
        return true;
      });
 
      if (itemsToRun.length > 0) {
        void this._executeBatch(itemsToRun);
      }
    }
  }
 
  private _scheduleNextBatchIfNeeded(key: string) {
    if (this._timeoutIds.has(key)) {
      // Already scheduled
      return;
    }
    // we wait 100ms to gather all requests, same as SPL batching
    const timeout = setTimeout(() => {
      const timeout = this._timeoutIds.get(key);
      if (timeout) {
        clearTimeout(timeout);
        this._timeoutIds.delete(key);
      }
      this._scheduleBatchWork(key);
    }, 100);
    this._timeoutIds.set(key, timeout);
  }
}
