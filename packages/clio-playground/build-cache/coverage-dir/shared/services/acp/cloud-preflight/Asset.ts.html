/**
 * Retrieves the children assets of a given directory from the Adobe Cloud Platform (ACP)
 * based on the provided access token and directory.
 *
 * @param {string} accessToken - The access token required for authentication with the Adobe Cloud Platform.
 * @param {ACPDirectory} directory - The ACPDirectory object representing the parent directory
 *                                   from which to retrieve the children (assets).
 *
 * @returns an array of ACPAsset objects representing the children assets of the given directory.
 *
 * @throws {Error} If there are issues with the authentication or the service request to the Adobe Cloud Platform.
 */
export async function acpGetChildren(
  accessToken: string,
  directory: ACPDirectory,
): Promise<Array<ACPAsset> | undefined> {
  const primayUrl = parseAcpBodyLink(directory.links, LinkRelation.PRIMARY);
 
  const { parsedBody: metadata } = await acpGetMetadata(
    accessToken,
    primayUrl!,
  );
 
  const { children } = metadata as Partial<{ children: Array<object> }>;
  if (!children) {
    return undefined;
  }
 
  return children.map((child: object) => {
    const childMetadata = child as Record<string, unknown>;
    const assetClass = childMetadata["repo:assetClass"] as ACPAssetClass;
    const assetId = childMetadata["repo:assetId"];
    const repoPath = childMetadata["repo:path"];
    const repoName = childMetadata["repo:name"];
    const repoId = childMetadata["repo:repositoryId"];
    const links = childMetadata._links;
    return {
      assetClass,
      assetId,
      repoPath,
      repoId,
      repoName,
      links,
    } as ACPAsset;
  });
}
 
/**
 * Retrieves an asset from the Adobe Cloud Platform (ACP) by its path,
 * searching recursively through the directory structure.
 *
 * @param {string} accessToken - The access token for authentication.
 * @param {string} path - The path of the asset to retrieve.
 * @param {ACPDirectory} asset - The starting directory to begin the recursive search.
 * @returns the found asset or undefined if not found.
 */
async function getAssetByPathDeep(
  accessToken: string,
  path: string,
  asset: ACPDirectory,
): Promise<ACPDirectory | undefined> {
  if (path === asset.repoPath) {
    return asset;
  }
 
  const children = await acpGetChildren(accessToken, asset);
  if (!children) {
    return undefined;
  }
 
  for (const child of children) {
    if (child.repoPath === path) {
      return child;
    }
 
    // the path is a sub path of child.repoPath
    if (path.startsWith(`${child.repoPath}/`)) {
      return getAssetByPathDeep(accessToken, path, child);
    }
  }
}
 
/**
 * Retrieves an asset from the Adobe Cloud Platform (ACP) based on the provided access token,
 * asset path, and the ACPDirectory representing the home directory.
 *
 * @param {string} accessToken - The access token required for authentication with the Adobe Cloud Platform.
 * @param {string} path - The path of the asset to be retrieved within the ACP.
 * @param {ACPDirectory} acpHome - The ACPDirectory representing the home directory in which the asset resides.
 *
 * @returns an ACPAsset object representing the retrieved asset from the ACP. 'undefined' is returned If the asset is not found.
 *
 * @throws {Error} If there are issues with the authentication or the service request to the Adobe Cloud Platform.
 */
export async function acpGetAssetByPath(
  accessToken: string,
  path: string,
  acpHome: ACPDirectory,
): Promise<ACPAsset | undefined> {
  // remove leading /
  path = path.replace(/^\/+/, "");
 
  let fullPath = "";
  if (acpHome.repoPath.endsWith("/")) {
    fullPath = `${acpHome.repoPath}${path}`;
  } else {
    fullPath = `${acpHome.repoPath}/${path}`;
  }
  return await getAssetByPathDeep(accessToken, fullPath, acpHome);
}
 
export async function acpGetDirectoryByPath(
  accessToken: string,
  path: string,
  acpHome: ACPDirectory,
): Promise<ACPDirectory | undefined> {
  const asset = await acpGetAssetByPath(accessToken, path, acpHome);
  return asset?.assetClass === ACPAssetClass.DIRECTORY ? asset : undefined;
}

function getAssetRenditionUrlFromCCStorage(
  assetManifestResponse: CCStorageManifest,
  assetId: string,
) {
  const assetComponent = assetManifestResponse.components.find(
    (component: { name: string; path: string }) =>
      component.name === MANIFEST_RENDITION_KEY,
  );
  const assetRenditionUrl = buildUrl(
    `${CCStorageConfig.renditionUrl}/${assetId}/${URL_PATHS.CONTENT}/${assetComponent?.path}/${URL_PATHS.VERSION_0}`,
    OLD_SHARED_CLOUD_API_QUERY_PARAMS,
  );
  return assetRenditionUrl;
}
 
function getReferenceImageUrlFromCCStorage(
  assetManifestResponse: CCStorageManifest,
  assetId: string,
  referenceImageRel: string,
) {
  const assetManifestInputChild = assetManifestResponse.children.find(
    (child) => child.path === INPUT_PATH_KEY,
  );
  const referenceImageComponent = assetManifestInputChild?.components.find(
    (component) => component.name === referenceImageRel,
  );
  const referenceImageUrl = referenceImageComponent
    ? buildUrl(
        `${CCStorageConfig.renditionUrl}/${assetId}/${URL_PATHS.CONTENT_INPUT}/${referenceImageComponent.path}/${URL_PATHS.VERSION_0}`,
        OLD_SHARED_CLOUD_API_QUERY_PARAMS,
      )
    : null;
  return referenceImageUrl;
}
 
export async function getAssetMetadataFromCCStorage(
  assetId: string,
  accessToken: string,
  locale: string | undefined,
): Promise<CCStorageAssetInfo> {
  try {
    const modelUrl = `${CCStorageConfig.modelUrl}/${assetId}`;
    // get Asset Metadata
    const assetModelResponse = await getCCStorageModel(modelUrl, accessToken);
    // parse model
    const modelJson = JSON.parse(
      assetModelResponse.custom.input[INPUT_MODEL_KEY],
    );
    modelJson.model.prompt =
      assetModelResponse.custom.input[MODEL_PROMPT_KEY]?.[`${locale}`] ??
      assetModelResponse.title;
 
    const isOwner = assetModelResponse.is_owner;
    const styleReferenceImageExists = modelJson?.model?.referenceImage;
    const structureReferenceImageExists =
      modelJson?.model?.structureReferenceImage;
    const manifestUrl = assetModelResponse._links.dcx_manifest.href;
    const assetManifestResponse = await getManifestCCStorage(
      manifestUrl,
      accessToken,
    );
    const assetRenditionUrl = getAssetRenditionUrlFromCCStorage(
      assetManifestResponse,
      assetId,
    );
    const referenceImage = styleReferenceImageExists
      ? getReferenceImageUrlFromCCStorage(
          assetManifestResponse,
          assetId,
          GUIDE_IMAGE_COMPONENT_NAME,
        )
      : null;
    const structureReferenceImage = structureReferenceImageExists
      ? getReferenceImageUrlFromCCStorage(
          assetManifestResponse,
          assetId,
          STRUCTURE_MATCH_COMPONENT_NAME,
        )
      : null;
 
    const dataModelVersion =
      assetModelResponse.custom[MODEL_INFO_KEY].dataModelVersion;
    const ccSharedCloudMetadata: CCStorageAssetInfo = {
      ...modelJson,
      cdnAccessToken: "",
      assetId,
      isOwner,
      imageUrl: assetRenditionUrl,
      dataModelVersion,
      referenceImage,
      structureReferenceImage,
    };
 
    return ccSharedCloudMetadata;
  } catch (err) {
    console.error(err);
    throw err;
  }
}
 
async function getCCStorageModel(
  modelUrl: string,
  accessToken: string,
): Promise<{
  title: string;
  custom: {
    input: {
      [INPUT_MODEL_KEY]: string;
      [MODEL_PROMPT_KEY]?: Record<string, string>;
    };
    [MODEL_INFO_KEY]: { dataModelVersion: string };
  };
  _links: { dcx_manifest: { href: string } };
  is_owner: boolean;
}> {
  const res = await fetch(modelUrl, {
    method: "GET",
    headers: {
      "X-Api-Key": CCStorageConfig.apiKey,
      ...(accessToken ? { Authorization: `Bearer ${accessToken}` } : {}),
    },
  });
  if (!res.ok) {
    throw new Error(`Cannot get CCStorageModel: ${JSON.stringify(res)}`);
  }
  return res.json();
}

export async function getCCStorageRendition(
  renditionUrl: string,
  accessToken: string,
): Promise<Blob> {
  let headers;
  // CDN asset access does not require headers,
  // it is publicly accessible.
  if (!isCDNStorageAsset(renditionUrl)) {
    headers = {
      "X-Api-Key": CCStorageConfig.apiKey,
      ...(accessToken ? { Authorization: `Bearer ${accessToken}` } : {}),
    };
  }
 
  const res = await fetch(renditionUrl, {
    method: "GET",
    headers,
  });
 
  return res.blob();
}
