const STORAGE_SCOPE = "hostApp:";
// Providing a hard-coded "text2Image" module id for now but you could imagine a future where CustomStylesManager
// could be scoped to any module. i.e if the user is in the TextEffects module,
// they should get only the models trained for TextEffects.
// We currently only train "Text2Image" models though, so this is fine for now.
const STYLES_STORAGE_LOCATION = "trainedStyles:text2image";
 
type CustomText2ImageStyle = Text2ImageStyle & {
  monitorURL?: string;
  ready?: boolean;
};
 
export type CustomText2ImageStyleData = Omit<
  Text2ImageStyleData,
  "popular" | "styles"
> & { styles: Array<CustomText2ImageStyle> };
 
/**
 * An observable interface for creating and using custom Text2Image styles.
 * Saves trained models to local storage and provides a way to monitor their training status.
 * https://git.corp.adobe.com/pages/Clio/clio-website/latest/apps/dreambooth.html#tyom-concept-finetuner
 */
export class CustomStylesManager {
  // Make the map observable so that we can keep the views up to date.
  private _styles = observable.map(new Map<string, CustomText2ImageStyle>());
  private _storage: Storage;
 
  constructor(private _splService: SPLServiceWrapper) {
    this._storage = createScopedStorage(STORAGE_SCOPE, localStorage);
    // Restore any previously trained styles.
    this._refreshStyles();
    // Check and update the status of any restored trained styles.
    for (const id of this._styles.keys()) {
      void this._monitorStyle(id);
    }
  }
 
  @computed
  get styleData(): CustomText2ImageStyleData {
    return {
      styles: Array.from(this._styles.values()),
      groups: [
        {
          id: TYOM_GROUP_ID,
          // In the future, when custom styles live on the server,
          // this group title would be localized along with the other style groups in fetchText2ImageStyles.
          title: "Your styles",
        },
      ],
    };
  }
 
  get customStyles() {
    return this._styles;
  }
 
  // Restore a map of trained styles from a previous session.
  private _refreshStyles() {
    const storageStr = this._storage.getItem(STYLES_STORAGE_LOCATION);
    Eif (!storageStr) {
      return;
    }
    this._styles.replace(JSON.parse(storageStr));
  }
 
  // Store the current map into local storage for a future session.
  private _saveStyles() {
    const storageStr = JSON.stringify(this._styles.toJSON());
    this._storage.setItem(STYLES_STORAGE_LOCATION, storageStr);
  }
 
  addStyle(name: string, { concept, monitorURL }: TYOMResult) {
    // We'll treat the concept as the style identifier.
    // Create a storage entry for the new style.
    // For now we only store the concept, name and monitorURL.
    // But in the future we may want to store more information about the style like the files used to train it, etc.
    this._styles.set(concept, {
      monitorURL,
      label: name,
      // The TYOM service uses its own text2Image model which injects the concept into the prompt using the $concept token.
      value: concept,
      group: TYOM_GROUP_ID,
      stylePrompt: "$concept",
      // Start with a default thumbnail.
      // Once training is complete, we'll generate a thumbnail using the new model!
      image: defaultStyleThumbnail,
    });
    this._saveStyles();
    // Start monitoring the style for usability.
    void this._monitorStyle(concept);
  }
 
  @action
  private _updateStyle(id: string, data: Partial<CustomText2ImageStyle>) {
    const style = this._styles.get(id);
    if (!style) {
      return;
    }
    this._styles.set(id, { ...style, ...data });
    this._saveStyles();
  }
 
  private async _monitorStyle(id: string, waitMinutes = 5): Promise<void> {
    const style = this._styles.get(id);
 
    if (!style?.monitorURL || style.ready) {
      if (style?.ready) {
        // When the style is ready, use it to generate its own thumbnail.
        void this._updateStyleThumbnail(id);
      }
      return;
    }
 
    // Check the status of the training and mark it as ready if it's done.
    const response = await fetch(style.monitorURL);
    const { status } = await response.json();
    this._updateStyle(id, { ready: status === "completed" });
 
    // Training takes about 10 minutes so check the status more frequently as we get closer to 10 minutes.
    await new Promise((resolve) =>
      setTimeout(resolve, 1000 * 60 * waitMinutes),
    );
    return this._monitorStyle(
      id,
      waitMinutes > 1 ? waitMinutes - 1 : waitMinutes,
    );
  }
 
  private async _updateStyleThumbnail(id: string): Promise<void> {
    // Wait for the SPLService to be ready.
    await when(() => !!this._splService.splStatus);
    const style = this._styles.get(id);
    if (!style || style.image !== defaultStyleThumbnail) {
      // Don't create a thumbnail if the style already has one.
      return;
    }
 
    const model: Text2ImageModel = {
      // Use the prompt that generates the preset Firefly style thumbnails.
      prompt: "hot air balloon",
      styles: { mode: "manual", value: [id] },
      seed: generateRandomSeed(),
      aspectRatio: "square" as ImageAspectRatio,
      advancedParams: {},
    };
 
    const { signal } = new AbortController();
    void (await this._splService.fetchStyleData(signal));
 
    const { image } = await this._splService.text2Image(
      ModuleId.Text2Image,
      model,
      "image/jpeg",
      signal,
    );
 
    // Get a smaller version of the image to save as thumbnail.
    const thumbnail = await getResizedBlob(image, 200, 200);
    if (thumbnail) {
      const dataUrl = await toDataURLFromBlob(thumbnail);
      this._updateStyle(id, { image: dataUrl });
    }
  }
}
