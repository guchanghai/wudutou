/**
 * Get basic manifest information for generated images.
 * Only exported for unit tests.
 * @param mimeType Mimetype of final output image
 * @returns Object containing manifest information
 *
 * @todo https://jira.corp.adobe.com/browse/CAI-4020
 * SDK has a type resolution mismatch, where the expected format is considered improper
 */
export function getManifestData(
  mimeType: string,
  isInference: boolean,
  module?: ModuleId,
): ManifestDefinition {
  const fullyTrainedModules = [ModuleId.Text2Image];
  const editedModules = [ModuleId.InPainting];
  const digitalSourceType =
    module && fullyTrainedModules.includes(module)
      ? "http://cv.iptc.org/newscodes/digitalsourcetype/trainedAlgorithmicMedia"
      : "http://cv.iptc.org/newscodes/digitalsourcetype/compositeWithTrainedAlgorithmicMedia";
  const actionType =
    module && editedModules.includes(module) ? "c2pa.edited" : "c2pa.created";
 
  const assertion: Assertion = {
    label: "c2pa.actions",
    data: {
      // @ts-expect-error - need to figure why SDK is complaining this isn't proper Assertion
      actions: [
        {
          action: actionType,
          softwareAgent: "Adobe Firefly",
          digitalSourceType: digitalSourceType,
        },
      ],
    },
  };
 
  if (isInference) {
    assertion.data.metadata = {
      "com.adobe.inference": isInference,
    };
  }
 
  const assertions: Array<Assertion> = [assertion];
 
  return {
    // Do not change claim_generator name: https://jira.corp.adobe.com/browse/CAI-3932
    // eslint-disable-next-line camelcase -- Manifest descriptor object takes camelcase key names
    claim_generator: "Adobe_Firefly",
    title: "Generated image",
    format: mimeType,
    assertions,
  };
}

export class CAIUtils {
  private _sdk: undefined | Promise<C2pa> = undefined;
 
  constructor(private _config?: Partial<C2paConfig>) {}
 
  private async _getSDK(): Promise<C2pa> {
    if (typeof this._sdk === "undefined") {
      const actualConfig = Object.assign({}, c2paConfig, this._config);
      this._sdk = createC2pa(actualConfig).catch((err) => {
        console.error("Error initializing C2PA SDK");
        // Throw error for downstream functions to pick up
        throw err;
      });
    }
    return this._sdk;
  }
 
  /**
   * Reads and validates C2PA data from an asset
   * @param asset The source of an asset to read
   * @returns The manifest information as a C2paReadResult (if found)
   */
  async readAsset(asset: Blob | CAIUrl): Promise<C2paReadResult | undefined> {
    const blob = await caiInputToBlob(asset);
 
    try {
      const c2paSdk = await this._getSDK();
      return c2paSdk.read(blob);
    } catch (err) {
      // TODO(rosca): Report to error reporting service, if available
      // Putting a console error for now so we know this condition was hit
      console.error(`Error reading C2PA data:`, err);
    }
  }
 
  /**
   * Gets mainfest guid of signed asset
   * @param asset The source of an asset to read
   * @returns string of the manifest guid
   */
  async getCAIManifestMapAndActiveKey(asset: Blob | CAIUrl): Promise<{
    manifestMap: ManifestMap | undefined;
    activeManifestId: string | undefined;
  }> {
    const readResult = await this.readAsset(asset);
    const manifestMap = readResult?.manifestStore?.manifests;
    const activeManifest = readResult?.manifestStore?.activeManifest;
    if (!activeManifest) {
      return {
        activeManifestId: undefined,
        manifestMap,
      };
    }
    let activeKey: string | undefined;
    const instanceId = activeManifest.instanceId;
    if (!manifestMap) {
      return {
        activeManifestId: undefined,
        manifestMap: undefined,
      };
    }
    Object.keys(manifestMap).forEach((key) => {
      if (manifestMap[key].instanceId === instanceId) {
        activeKey = key;
      }
    });
    return {
      activeManifestId: activeKey,
      manifestMap,
    };
  }
 
  /**
   * Checks if the data value of an assertion is equal to the expected value in all manifests
   * @param asset The source of an asset to read
   * @param assertionLabel The label of the assertion containing the data key to check
   * @param dataKey The manifest data key to check
   * @param expectedValue The value that the data key's value should be equal to
   * @returns True if the data value is equal to the expected value, false otherwise, undefined if the assertion does not exist
   */
  async checkAssertionDataValueAllManifests(
    asset: Blob | CAIUrl,
    assertionLabel: string,
    dataKeys: Array<string>,
    expectedValues: Array<string>,
  ): Promise<boolean | undefined> {
    const readResult = await this.readAsset(asset);
    const manifestMap = readResult?.manifestStore?.manifests;
    let valuesEqual = true;
    let trainingAssertions: Array<Assertion> = [];
 
    if (manifestMap) {
      for (const key in manifestMap) {
        if (Object.prototype.hasOwnProperty.call(manifestMap, key)) {
          trainingAssertions = trainingAssertions.concat(
            // @ts-expect-error - SDK is complaining this isn't proper Assertion type
            manifestMap[key].assertions.get(
              // @ts-expect-error - need to figure why SDK is complaining this isn't proper Assertion
              assertionLabel,
            ),
          );
        }
      }
    }
 
    if (trainingAssertions.length === 0) {
      return undefined;
    }
 
    valuesEqual = trainingAssertions.every((assertion) => {
      const keyAssertion = dataKeys.every((key) => {
        // @ts-expect-error - SDK is complaining about Assertion type
        if (expectedValues.includes(assertion.data.entries[key].use)) {
          return true;
        }
      });
 
      return keyAssertion;
    });
 
    return valuesEqual;
  }
 
  /**
   * Reads C2PA data from an asset and determines if the `c2pa.ai_inference` tag is allowed
   * @param asset  The source of an asset to read
   * @returns false if the `c2pa.ai_inference` tag is not allowed or constrained, true otherwise
   */
  async isInferenceAllowed(asset: Blob | CAIUrl): Promise<boolean> {
    const valueMatchesNotAllowed =
      await this.checkAssertionDataValueAllManifests(
        asset,
        "c2pa.training-mining",
        ["c2pa.ai_inference"],
        ["notAllowed", "constrained"],
      );
    return !valueMatchesNotAllowed;
  }

  /**
   * Signs and embeds C2PA data inside an asset
   * @param data The data of the asset to sign, optionally the original image
   * @param authToken Authentication key for signing
   * @param module Optional Module id of the image to sign
   * @param referenceImageStr Optional style reference image string
   * @param structureReferenceImageStr Optional structure reference image string
   * @param progress Optional callback for progress UI
   * @returns The signed asset with embedded metadata as a Blob
   */
  async signAsset(
    image: Blob | CAIUrl,
    authToken: string,
    module?: ModuleId,
    includeThumbnail = true,
    referenceImageStr?: string,
    structureReferenceImageStr?: string,
    progress?: SignProgressCallback,
  ): Promise<Blob> {
    const [blob, c2paSdk] = await Promise.all([
      caiInputToBlob(image),
      this._getSDK(),
    ]);
 
    const manifestInfo = await this.buildManifest(
      blob,
      c2paSdk,
      module,
      referenceImageStr,
      structureReferenceImageStr,
    );
 
    const { embeddedAsset } = await c2paSdk.sign(blob, manifestInfo, {
      authToken,
      storageLocations: ["cloud", "embedded"],
      onProgress: (input) => progress?.(input.value),
      ...(!includeThumbnail && { thumbnail: includeThumbnail }),
    });
 
    return embeddedAsset;
  }
 
  /**
   * Signs and embeds C2PA data inside an asset
   * @param blob The blob of the asset to sign
   * @param module Optional Module id of the image to sign
   * @param referenceImageStr Optional style reference image string
   * @param structureReferenceImageStr Optional structure reference image string
   * @returns Promise manifest info
   */
  async buildManifest(
    blob: Blob,
    c2paSdk: C2pa,
    module?: ModuleId,
    referenceImageStr?: string,
    structureReferenceImageStr?: string,
  ): Promise<ManifestBuilder> {
    const isInference =
      Boolean(referenceImageStr) || Boolean(structureReferenceImageStr);
    const manifestData = getManifestData(blob.type, isInference, module);
 
    const manifestInfo = c2paSdk.createManifest(manifestData);
 
    if (referenceImageStr) {
      await this.addIngredientToManifest(
        c2paSdk,
        referenceImageStr,
        "Reference Image",
        manifestInfo,
      );
    }
 
    if (structureReferenceImageStr) {
      await this.addIngredientToManifest(
        c2paSdk,
        structureReferenceImageStr,
        "Composition Reference Image",
        manifestInfo,
      );
    }
 
    return manifestInfo;
  }
 
  async addIngredientToManifest(
    c2paSdk: C2pa,
    imageSrc: string,
    title: string,
    manifestInfo: ManifestBuilder,
  ) {
    const blobRes = await fetch(imageSrc);
    const imageBlob = await blobRes.blob();
 
    const storableIngredient = await c2paSdk.createIngredient({
      input: imageBlob,
      title: title,
    });
 
    // @ts-expect-error: ignore because of cai sdk limitation
    delete storableIngredient.ingredient.thumbnail;
    // @ts-expect-error: ignore because of cai sdk limitation
    delete storableIngredient.ingredient.manifest_data;
    storableIngredient.resources = [] as unknown as Record<string, ArrayBuffer>;
    storableIngredient.ingredient.relationship = "inputTo";
 
    await manifestInfo.addIngredient(storableIngredient);
  }
 
  /**
   * Verifies if the asset is signed.
   * @param asset The asset to verify
   * @returns true if the asset is signed.
   */
  async isAssetSigned(asset: Blob | CAIUrl): Promise<boolean> {
    const assetInfo = await this.readAsset(asset);
    const signatureInfo =
      assetInfo?.manifestStore?.activeManifest.signatureInfo;
    return Boolean(signatureInfo);
  }
 
  /**
   * Verifies if the asset is signed by Firefly service by checking cert_serial_number
   * @param asset The asset to verify
   * @returns true if the asset is signed by Firefly service.
   */
  async verifyAssetSignature(asset: Blob | CAIUrl): Promise<boolean> {
    const assetInfo = await this.readAsset(asset);
 
    const signatureInfo =
      assetInfo?.manifestStore?.activeManifest.signatureInfo;
    if (signatureInfo) {
      /* eslint-disable-next-line @typescript-eslint/no-explicit-any -- cast to access cert_serial_number. */
      const certSerialNumber = (signatureInfo as any).cert_serial_number;
      if (VALID_CERT_SERIAL_NUMBERS.includes(certSerialNumber)) {
        return true;
      }
    }
 
    return false;
  }
