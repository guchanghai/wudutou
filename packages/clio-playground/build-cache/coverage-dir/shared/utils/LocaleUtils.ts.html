/**
 * Normalize ZH locale if exists
 *
 * Examples:
 *  - zh-Hans-CN -> zh-CN
 *  - zh-Hant-TW -> zh-TW
 *  - en_US -> en_US
 *
 * @returns normalized locale.
 */
export function normalizeIfZHLocale(locale: string) {
  const lowerCaseLocale = locale.toLowerCase();
  if (lowerCaseLocale in ZH_LOCALE_MAP) {
    return ZH_LOCALE_MAP[lowerCaseLocale];
  }
 
  // This is a fallback to check if there is a locale that starts with "zh"
  // that is not part of the ZH_LOCALE_MAP. If it is, we will return the
  // normalized locale, otherwise we will fallback into english.
  if (lowerCaseLocale.startsWith("zh")) {
    if (lowerCaseLocale.includes("hans") || lowerCaseLocale.includes("hant")) {
      // replace hans and hant with "-" symbol
      return locale.replace(/((-|_)(hans|hant)(-|_))/gi, "_");
    }
 
    // we don't know this locale, we fallback into english
    return DEFAULT_LOCALE_UNDERSOCRED;
  }
 
  return locale;
}
