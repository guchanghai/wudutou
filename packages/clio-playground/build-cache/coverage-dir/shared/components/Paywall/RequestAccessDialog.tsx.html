import { useCallback, useEffect, useState } from "react";
import { useAnalyticsContext } from "shared/contexts/AnalyticsContext";
import { useModuleConfig } from "shared/hooks/useModuleConfig";
import {
  ClickUpgradeSourceType,
  PageSource,
} from "shared/types/AnalyticsTypes";
import { getEventData } from "shared/utils/EventMessageUtils";
import { openTo } from "shared/utils/Navigate";
 
import { ModalIframe } from "../ModalIframe/ModalIframe";
 
type Props = {
  onClose?: () => void;
  onAccessProvided?: () => void;
  entryUrl?: string;
};
 
export type CommerceReturnData = {
  returnURL: string;
  returnURLParams?: Array<string>;
};
 
enum ACRSClosingStatus {
  ACCESS_PROVIDED = "ACCESS_PROVIDED",
  ACCESS_PENDING = "ACCESS_PENDING",
  ACCESS_REQUESTED = "ACCESS_REQUESTED",
  CANCELLED = "CANCELLED",
  FAILED = "FAILED",
}
 
/**
 * Component to show request access iframe and handle logic for access events.
 * Pls refer to https://developers.corp.adobe.com/abp/ACRS/docs/user_clients.md for message handlers.
 */
export function RequestAccessDialog({
  onClose,
  onAccessProvided,
  entryUrl,
}: Props) {
  const [width, setWidth] = useState<number>(0);
  const [height, setHeight] = useState<number>(0);
  const { analytics } = useAnalyticsContext();
  const moduleConfig = useModuleConfig();
 
  // We know this if the module config is not present
  const isHomePage = !moduleConfig;
 
  const handleOnMessage = useCallback((event: MessageEvent) => {
    // verify that the message is from the request access iframe
    if (entryUrl && event.origin !== new URL(entryUrl).origin) {
      return;
    }
 
    try {
      const eventData = getEventData(event);
 
      if (eventData.app !== "acrs-request-access") return;
 
      const { data } = eventData;
      // pls note that some methods are currently blank,
      // still want to have them for adding analytics in the future.
      if (eventData.type === "OpenURL") {
        if (
          eventData.subType === "EXTERNAL" ||
          eventData.subType === "SWITCH"
        ) {
          openTo(data.externalUrl);
        }
      } else if (eventData.type === "System") {
        switch (eventData.subType) {
          case "AppLoaded":
            break;
          case "Resize": {
            const newWidth = Math.min(data.width, window.innerWidth - 50);
            const newHeight = Math.min(data.height, window.innerHeight);
 
            setWidth(newWidth);
            setHeight(newHeight);
            break;
          }
 
          case "Close": {
            if (data.status === ACRSClosingStatus.ACCESS_REQUESTED) {
              analytics.recordClickRequestUpgradeEvent(
                isHomePage ? PageSource.home : moduleConfig.analyticsId,
                // TODO: Get help to get this from event data
                // https://jira.corp.adobe.com/browse/FIREFLY-7050
                "",
                ClickUpgradeSourceType.Banner,
              );
            } else if (data.status === ACRSClosingStatus.ACCESS_PROVIDED) {
              console.log("Requested Access is granted");
              onAccessProvided?.();
            } else if (data.status === ACRSClosingStatus.ACCESS_PENDING) {
              console.log("Requested Access is pending");
            }
            clear();
            break;
          }
          default:
        }
      }
    } catch (e) {
      console.log("caught Error:", e);
    }
  }, []);
 
  useEffect(() => {
    window.addEventListener("message", handleOnMessage);
    return () => {
      window.removeEventListener("message", handleOnMessage);
    };
  }, []);
 
  const clear = () => {
    onClose?.();
  };
 
  if (!entryUrl) {
    return null;
  }
 
  return (
    <ModalIframe
      onClose={() => {
        clear();
      }}
      url={entryUrl}
      width={width}
      minWidth={width}
      maxWidth={width}
      height={height}
      minHeight={height}
      maxHeight={height}
      data-testid="request-access-dialog"
    />
  );
}
