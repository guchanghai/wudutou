import {
  listPublishedAssetsForUser,
  listPublishedPublicAssets,
} from "shared/services/community/CommunityService";
import {
  AssetState,
  type UGCEntry,
} from "shared/services/community/CommunityTypes";
import { ModuleId } from "shared/types/ModuleConfig";
 
export interface SubmissionsResponse {
  assets: Array<UGCSubmission>;
  nextCursor: Cursor | null;
}
 
export interface UGCSubmission {
  id: string;
  key: string;
  urn: string;
  manifest: string;
  /**
   * The module type of the submission.
   */
  type: ModuleId;
  thumbnail: string;
  title: string;
  state: AssetState;
  aspectRatio: number;
  prompt?: string;
  liked: boolean;
  likeCount: number;
  owner: {
    displayName: string;
    id: string;
    avatarUrl: string;
  };
  promptLocale: string;
}
 
export interface Cursor {
  state?: AssetState;
  apiCursor: string;
}
 
export const RENDITION_FORMAT = "jpg";
export const RENDITION_SIZE = "512";
export const RENDITION_DIMENSION = "width";
 
function convertStringToAssetState(state: string): AssetState {
  switch (state.toUpperCase()) {
    case "PENDING":
      return AssetState.PENDING;
    case "APPROVED":
      return AssetState.APPROVED;
    case "REJECTED":
      return AssetState.REJECTED;
    default:
      return AssetState.PENDING;
  }
}
 
function convertUGCEntryToUGCSubmissions(
  item: UGCEntry,
  source: string,
  locale?: string,
): UGCSubmission {
  let aspect =
    (item._embedded.artwork
      ? item._embedded.artwork.width
      : item._links.rendition.max_width) /
    (item._embedded.artwork
      ? item._embedded.artwork.height
      : item._links.rendition.max_height);
  aspect = !isNaN(aspect) ? aspect : 1;
 
  const avatarUrl = item._embedded.owner._links.images[0]
    ? item._embedded.owner._links.images[0].href
    : "";
 
  return {
    id: item.id,
    key: `${item.id}-${(Math.random() + 1).toString(36).substring(5)}`,
    manifest: item._links.dcx_manifest.href,
    type: ModuleId.Text2Image,
    thumbnail: item._links.rendition.href
      .replace("{format}", RENDITION_FORMAT)
      .replace("{dimension}", RENDITION_DIMENSION)
      .replace("{size}", RENDITION_SIZE),
    title:
      source === "userSubmissions"
        ? item.custom?.input["firefly#originalPrompt"] ?? item.title
        : item.custom?.input["firefly#prompts"]?.[`${locale}`] ?? item.title,
    state: item.state
      ? convertStringToAssetState(item.state)
      : AssetState.APPROVED,
    aspectRatio: aspect,
    liked: item.liked,
    likeCount: item.stats.like_count,
    owner: {
      id: item._embedded.owner.id,
      displayName: item._embedded.owner.display_name,
      avatarUrl,
    },
    urn: item.urn,
    promptLocale: item.custom?.input["firefly#locale"] ?? "",
  };
}
 
/**
 * Retrieves a paginated list of user submissions sorted by their approval status.
 * Initially fetches approved assets, followed by pending and rejected assets.
 * This function ensures that the total number of assets in the response does not
 * exceed the specified page size.
 *
 * @param {string} accessToken - The access token for authentication with the API.
 * @param {number} pageSize - The number of assets to be returned in each page of the response.
 * @param {Cursor} [cursor] - An optional cursor indicating the state and position from which
 *                            to continue fetching assets. If provided, the function resumes
 *                            fetching from the specified state and position. If omitted, fetching
 *                            begins with approved assets.
 * @returns {Promise<UserSubmissionsResponse>} An object containing an array of assets up to the
 *                                             specified page size and a cursor for the next page,
 *                                             if more assets are available.
 */
export async function getUserSubmissions({
  accessToken,
  pageSize,
  cursor,
  schemaValidationEnabled,
}: {
  accessToken: string;
  pageSize: number;
  cursor?: Cursor;
  schemaValidationEnabled: boolean;
}): Promise<SubmissionsResponse> {
  let assets: Array<UGCEntry> = [];
  let nextCursor: Cursor | null = null;
  let state: AssetState = AssetState.APPROVED;
  let cursorURL: string | undefined = cursor?.apiCursor;
 
  // If a cursor is provided, transfer its state.
  if (cursor?.state) {
    state = cursor.state;
  }
 
  while (assets.length < pageSize) {
    const response = await listPublishedAssetsForUser(
      accessToken,
      state,
      schemaValidationEnabled,
      pageSize,
      cursorURL,
    );
 
    assets = assets.concat(response._embedded.assets);
 
    if (response._links?.next) {
      cursorURL = response._links.next.href;
      nextCursor = { state, apiCursor: cursorURL };
    } else if (state === AssetState.APPROVED) {
      state = AssetState.PENDING;
      cursorURL = undefined;
    } else if (state === AssetState.PENDING) {
      state = AssetState.REJECTED;
      cursorURL = undefined;
    } else {
      // We stop if there is no cursor
      nextCursor = null;
      break;
    }
 
    if (assets.length >= pageSize) break;
  }
 
  if (assets.length < pageSize) {
    nextCursor = null;
  }
 
  return {
    assets: assets.map((item) =>
      convertUGCEntryToUGCSubmissions(item, "userSubmissions"),
    ),
    nextCursor,
  };
}
 
export async function getPublicSubmissions({
  pageSize,
  locale,
  cursor,
  accessToken,
  schemaValidationEnabled,
}: {
  pageSize: number;
  locale: string;
  cursor?: Cursor;
  accessToken?: string;
  schemaValidationEnabled: boolean;
}): Promise<SubmissionsResponse> {
  let assets: Array<UGCEntry> = [];
  let nextCursor: Cursor | null = null;
  let cursorURL: string | undefined = cursor?.apiCursor;
 
  while (assets.length < pageSize) {
    const response = await listPublishedPublicAssets(
      pageSize,
      schemaValidationEnabled,
      cursorURL,
      accessToken,
    );
    assets = assets.concat(response._embedded.assets);
 
    if (response._links?.next) {
      cursorURL = response._links.next.href;
      nextCursor = { apiCursor: cursorURL };
    } else {
      // We stop if there is no cursor
      nextCursor = null;
      break;
    }
 
    if (assets.length >= pageSize) break;
  }
 
  if (assets.length < pageSize) {
    nextCursor = null;
  }
 
  return {
    assets: assets.map((item) =>
      convertUGCEntryToUGCSubmissions(item, "gallery", locale),
    ),
    nextCursor,
  };
}
