  const {
    uid = useId(),
    description,
    thumbnail,
    model,
    seed,
    generationMetadata,
    type,
    download,
    downloadMimeType = "image/png",
    showContentCredentials,
    onUsePrompt,
    actionButtonStr,
    blurBackgroundOnOversize,
    width,
    height,
    aspectRatio,
    style,
    excludeFromTabOrder = false,
    hideDescription = false,
    hideActionButton = false,
    activeSample,
    setActiveSample,
    startSlideShow,
    endSlideShow,
    video,
    lightOverlay = video !== undefined,
    actionButtonLocation = "bottom-right",
    actionButtonIcon,
    ...rest

  const [isActive, setIsActive] = useState<boolean>(false);
  const [isEscHidden, setIsEscHidden] = useState<boolean>(false);
  const { analytics } = useAnalyticsContext();
  const { getStyleZeroPresets } = useCDNAssetContext();
  const [isImageLoaded, setIsImageLoaded] = useState(false);
  const exportUtils = useExportUtils();
  const intl = useIntl();
  const isMobile = useIsMobile();
  const touchDevice = isTouchDevice();
  const isIOSPlatform = getMobileOSType() === MobileOSType.IOS;
  const { isIndemnified, isPremium } = useEntitlementContext();
  const indemnifyAnalytics = useIndemnification();
  const textDescriptionRef = useRef<HTMLDivElement>(null);
  const textContainerRef = useRef<HTMLDivElement>(null);
  const [isAnimated, setIsAnimated] = useState<boolean>(false);
  const [animateSpeed, setAnimateSpeed] = useState<number>(0);
  const { showModalDownloadView } = useModalDownloadContext();
  const clickFunction = function (event: Event) {
    const isActiveSample = uid === activeSample;
    // Additional check for touch devices that also have a mouse pointer.
    const isPointer = (event as PointerEvent).pointerType === "mouse";
    touchDevice && !isPointer && !isActiveSample
      ? setActiveSample?.(uid)
      : onUsePrompt();
  };
  const toast = useToastContext();
  const caiErrorMessage: React.ReactNode = (
    <p>
      {intl.formatMessage(caiErrorMessages.toastHeader)}
      <br />
      {intl.formatMessage(caiErrorMessages.toastDescription)}
    </p>
  );
 
  const sampleRef = useRef<HTMLDivElement | null>(null);
 
  // Video related states
  const [videoLoaded, setVideoLoaded] = useState<boolean>(false);
  const videoRef = useRef<HTMLVideoElement>(null);
 
  // Only apply this memory optimizaiton on mobile.
  // Release the thumbnail memory when it is out of view.
  const [isOutOfView, setIsOutOfView] = useState(isMobile);
 
  async function handleDownload() {
    if (!download) {
      return;
    }
 
    const { signAsset, utils, showSigningErrorToast } = caiStore;
    const moduleIdEnum = type as ModuleId;
    const moduleConfig = registry.getModuleConfig(moduleIdEnum);
    const { getAnalyticsFavoritesPayload } = moduleConfig;
    const payload =
      getAnalyticsFavoritesPayload &&
      (await getAnalyticsFavoritesPayload(model));
    let styleReferenceImage: StyleReferenceImage | undefined;
    let styleReferenceImageSrc: string | undefined;
    if (payload) {
      styleReferenceImage = payload.referenceImage;
    }
    const controller = new AbortController();
    try {
      const styleZeroPresets = await getStyleZeroPresets(controller);
      if (styleReferenceImage && styleZeroPresets) {
        const previewImage = await getPreviewImage(
          styleReferenceImage,
          styleZeroPresets,
        );
        styleReferenceImageSrc = previewImage || undefined;
      }
    } catch (e) {
      console.error(`Error fetching style zero presets`, e);
    }
 
    const needWatermark = !isPremium;
    const moduleId = type && getModuleConfig(type).id;
    /*
    const isTextEffectsModule =
      type && getModuleConfig(type).id === (ModuleId.TextEffects as string);
    const isTextToImageModule =
      type && getModuleConfig(type).id === (ModuleId.Text2Image as string);
    */
    const blob = await toImageBlobFromURL(download, downloadMimeType);
    // always indemnify the original asset.
    if (isIndemnified) {
      indemnifyAnalytics(
        blob,
        downloadMimeType,
        FeedbackId.Text2Image,
        QuickActionSubtype.Download,
      );
    }
    try {
      let isAssetSigned = await utils.isAssetSigned(blob);
      let assetToDownload = blob;
      if (needWatermark) {
        const img = await exportUtils.applyWatermark(
          blob,
          downloadMimeType,
          needWatermark,
        );
        assetToDownload = await toBlobFromURL(img);
        isAssetSigned = false;
      }
 
      let signedAsset;
      if (!isAssetSigned) {
        // Sign the image with Content Authenticity credentials
        const { err, result } = await signAsset({
          image: assetToDownload,
          moduleId,
          referenceImage: {
            styleReferenceImage: styleReferenceImageSrc,
          },
          viewType: QuickActionSubtype.Download,
          generationMetadata,
        });
        if (err) {
          showSigningErrorToast(err, toast.show, caiErrorMessage);
        }
        signedAsset = result;
      } else {
        signedAsset = assetToDownload;
      }
 
      if (isIOSPlatform && moduleId === (ModuleId.Text2Image as string)) {
        const imageUrl = await toDataURLFromBlob(signedAsset);
        showModalDownloadView({ imageUrl, aspectRatio: "square" });
      } else {
        const urlObject = URL.createObjectURL(signedAsset);
        const filename = getDownloadedImageName(
          downloadMimeType,
          seed ?? 0,
          description,
        );
        exportUtils.downloadObjectUrlAsFile(urlObject, filename);
        if (type !== undefined && seed !== undefined) {
          analytics.recordFeedbackQuickAction({
            eventSubtype: QuickActionSubtype.Download,
            feedbackId: getModuleConfig(type).analyticsId,
            result: urlObject,
            source: FeedbackUIViewType.Normal,
            seed,
          });
        }
      }
    } catch (e) {
      console.error(`Error downloading favorite`, e);
    }
  }
 
  const tryPrompt = intl.formatMessage({
    id: "@clio/playground:inspire-sample-try-prompt-aria-label",
    defaultMessage: "View samples",
  });
 
  const tryPromptMobile = intl.formatMessage({
    id: "@clio/playground:inspire-sample-try-prompt-aria-label-mobile",
    defaultMessage: "View",
  });
 
  const downloadText = intl.formatMessage({
    id: "@clio/playground:inspire-sample-download-aria-label",
    defaultMessage: "Download",
  });
 
  useEffect(() => {
    const textContainer = textContainerRef.current;
    const textDescription = textDescriptionRef.current;
 
    if (startSlideShow && isActive) {
      startSlideShow();
    } else if (endSlideShow) {
      endSlideShow();
    }
 
    if (!textContainer || !textDescription || !isActive) {
      return;
    }
 
    // we need to wait for the text to be rendered before we can calculate the height
    const containerHeight = textContainer.clientHeight;
    const descriptionHeight = textDescription.clientHeight;
    setIsAnimated(containerHeight < descriptionHeight);
    if (!(containerHeight < descriptionHeight)) {
      return;
    }
    const animationRate = 1.5; // 1.5 seconds per line
    // line-height of var(--spectrum-global-dimension-font-size-300)
    // is 16px, but we need to account for the line-height of different locales
    // using line height multiplier at https://spectrum.corp.adobe.com/page/body/
    let lineHeight;
    switch (intl.locale) {
      case "ja-JP":
      case "zh-Hans-CN":
      case "zh-Hant-TW":
      case "ko-KR":
        lineHeight = 16 * 1.7;
        break;
      default:
        lineHeight = 16 * 1.5;
        break;
    }
    setAnimateSpeed(Math.round(descriptionHeight / lineHeight) * animationRate);
  }, [isActive]);
 
  useEffect(() => {
    if (uid === activeSample) {
      setIsActive(true);
    } else {
      setIsActive(false);
    }
  }, [activeSample]);
 
  useEffect(() => {
    // Only apply this memory optimization on mobile.
    if (!isMobile) {
      setIsOutOfView(false);
      return;
    }
    const sampleView = sampleRef.current;
    if (!sampleView) {
      return;
    }
 
    if ("IntersectionObserver" in window) {
      const observer = new IntersectionObserver(
        (entries) => {
          let inView = false;
          entries.forEach((entry) => {
            if (entry.isIntersecting) {
              inView = true;
            }
          });
          setIsOutOfView(!inView);
        },
        // Add more margin to the rootMargin to make sure the image is loaded faster.
        {
          rootMargin: "200%",
          threshold: 0,
        },
      );
 
      observer.observe(sampleView);
 
      return () => {
        observer.disconnect();
      };
    }

      behavior: "smooth",
      block: "nearest",
      inline: "nearest",
    });
    if (!touchDevice) {
      if (activeSample) {
        setActiveSample?.(uid);
      } else {
        setIsActive(true);
      }
    }
    setIsEscHidden(false);
  };
 
  const onBlurCapture = (evt: FocusEvent) => {
    if (parentArticleRef.current?.contains(evt.relatedTarget as Node)) {
      return;
    }
    if (!touchDevice) {
      if (activeSample) {
        setActiveSample?.(undefined);
      } else {
        setIsActive(false);
      }
    }
  };
 
  const onKeyDown = (evt: KeyboardEvent) => {
    const target = evt.target as HTMLElement;
    switch (evt.key) {
      case "Escape":
        if (
          target.classList.contains(css["download-button"]) ||
          target.classList.contains(css["action-button"])
        ) {
          parentArticleRef.current?.focus();
        }
        setIsEscHidden(!isEscHidden);
        break;
      case "Enter":
      case " ":
        clickFunction(evt);
        break;
    }
  };
 
  useEffect(() => {
    parentArticleRef.current = sampleRef.current?.closest(
      'article, [role="article"]',
    );
    parentArticleRef.current?.addEventListener("focus", onFocusCapture, true);
    parentArticleRef.current?.addEventListener("blur", onBlurCapture, true);
    parentArticleRef.current?.addEventListener("keydown", onKeyDown);
    return () => {
      parentArticleRef.current?.removeEventListener(
        "focus",
        onFocusCapture,
        true,
      );
      parentArticleRef.current?.removeEventListener(
        "blur",
        onBlurCapture,
        true,
      );
      parentArticleRef.current?.removeEventListener("keydown", onKeyDown);
    };
  });
 
  const playVideo = async (videoEle: HTMLVideoElement | null) => {
    if (videoEle) {
      try {
        await videoEle.play();
      } catch (error) {
        console.error(error);
      }
    }
  };
 
  const onVideoMouseEnter = (
    e: MouseEvent | React.FocusEvent<HTMLDivElement>,
  ) => {
    const videoEle = videoRef.current;
    if (videoEle) {
      e.stopPropagation();
      void playVideo(videoEle);
    }
  };
 
  const onVideoMouseLeave = (
    e: MouseEvent | React.FocusEvent<HTMLDivElement>,
  ) => {
    const videoEle = videoRef.current;
    if (videoEle) {
      e.stopPropagation();
      videoEle.pause();
      videoEle.currentTime = 0;
    }
  };
 
  const imgId = `img${uid}`;
 
  const holderId = description
    ? hideDescription
      ? imgId
      : `description${uid}`
    : undefined;
 
  const tryPromptButtonId = `try-prompt-button${uid}`;
 
  const tryPromptButtonLabel = isMobile ? tryPromptMobile : tryPrompt;
 
  const tryPromptButtonAriaLabelledby =
    holderId && `${actionButtonStr ?? tryPromptButtonLabel}` !== description
      ? `${tryPromptButtonId} ${holderId}`
      : undefined;
 
  return (
    <div
      {...rest}
      style={{ ...style, aspectRatio, width, height }}
      className={css["inspire-holder"]}
      role="figure"
      aria-label={hideDescription ? description : undefined}
      aria-labelledby={hideDescription ? undefined : holderId}
      ref={sampleRef}
      onClick={(e) => {
        clickFunction(e.nativeEvent);
      }}
      onFocusCapture={(evt: React.FocusEvent<HTMLDivElement>) => {
        evt.currentTarget.scrollIntoView({
          behavior: "smooth",
          block: "nearest",
          inline: "nearest",
        });
        onVideoMouseEnter(evt);
      }}
      onBlurCapture={(evt: React.FocusEvent<HTMLDivElement>) => {
        onVideoMouseLeave(evt);
      }}
      onPointerEnter={(e: React.PointerEvent) => {
        if (
          e.pointerType !== "touch" &&
          /* Prevents mouse enter from triggering when the mouse is over ContentCredentialsOverlay dialog */
          e.currentTarget.contains(e.target as Node)
        ) {
          setIsEscHidden(false);
          if (activeSample) {
            setActiveSample?.(uid);
          } else {
            setIsActive(true);
          }
          onVideoMouseEnter(e);
        }
      }}
      onPointerLeave={(e: React.PointerEvent) => {
        if (e.pointerType !== "touch") {
          if (activeSample) {
            setActiveSample?.(undefined);
          } else {
            setIsActive(false);
          }
          onVideoMouseLeave(e);
        }
      }}
    >
      <ImageWithPlaceholder
        id={imgId}
        alt={description}
        aria-labelledby={hideDescription ? undefined : holderId}
        src={isOutOfView ? "" : thumbnail}
        objectFit="cover"
        useLazyLoad={true}
        blurBackgroundOnOversize={blurBackgroundOnOversize}
        onLoad={() => setIsImageLoaded(true)}
        className={classnames({
          [css.loaded]: !isImageLoaded, // Show the image once it's loaded
          [css.hidden]: videoLoaded, // Hide the image once the video is loaded
        })}
        enableIntersectionObserver={false}
      />
      {video && (
        <video
          slot="cover-photo"
          loop
          muted
          preload="none" // to prevent data from loading so it doesn't affect the performance
          playsInline
          ref={videoRef}
          className={classnames(css.loaded, {
            [css.hidden]: !videoLoaded, // Hide the video until it's loaded
          })}
          onCanPlay={() => {
            // Known issue: If laptop in low power mode Safari will require a user click interaction to play the video
            setVideoLoaded(true);
          }}
        >
          <source src={video} type="video/mp4" />
        </video>
      )}
      {showContentCredentials && isImageLoaded && (
        <ContentCredentialsOverlay
          imageUrl={isOutOfView ? "" : thumbnail}
          aria-labelledby={holderId}
          excludeFromTabOrder={excludeFromTabOrder}
        />
      )}
      <Flex
        className={classnames(css["inspire-sample-overlay"], {
          [css["inspire-sample-overlay-selected"]]: isActive,
          [css["inspire-sample-overlay-hidden"]]: isEscHidden,
        })}
        style={{
          background: lightOverlay
            ? actionButtonLocation === "top-left"
              ? "linear-gradient(0deg, rgba(0,0,0,0.1) 50%, rgba(0,0,0, 0.5))"
              : "linear-gradient(rgba(0,0,0,0.1) 50%, rgba(0,0,0, 0.5))"
            : "#000000a5",
        }}
        direction="column"
      >
        {!hideDescription && (
          <div
            ref={textContainerRef}
            className={classnames(css["inspire-sample-description-container"], {
              [css["inspire-sample-description-container-animated"]]:
                isAnimated,
            })}
            style={{ "--animation-speed": `${animateSpeed}s` } as CSSProperties}
            role="presentation"
            hidden={!isActive}
          >
            <figcaption
              className={classnames(css["inspire-sample-description"], {
                [css.isAnimated]: isAnimated,
              })}
              id={holderId}
              ref={textDescriptionRef}
            >
              {description}
            </figcaption>
            {isAnimated && (
              // need duplicated description to create infinite loop for animation
              <div
                className={classnames(css["inspire-sample-description"], {
                  [css.isAnimated]: isAnimated,
                })}
                role="presentation"
                aria-hidden="true"
              >
                {description}
              </div>
            )}
          </div>
        )}
        <Theme color="light">
          {download && (
            <IconButton
              icon={<Download />}
              label={downloadText}
              className={css["download-button"]}
              showTooltip={false}
              onClick={(e) => {
                void handleDownload();
                e.stopPropagation();
              }}
              tabIndex={excludeFromTabOrder ? -1 : 0}
            />
          )}
 
          {!hideActionButton && (
            <div className={css["inspire-sample-action"]}>
              <Button
                className={classnames(css["action-button"], {
                  [css["action-button-top-left"]]:
                    actionButtonLocation === "top-left",
                  [css["action-button-bottom-right"]]:
                    actionButtonLocation === "bottom-right",
                })}
                variant="secondary"
                treatment="fill"
                onClick={onUsePrompt}
                id={tryPromptButtonId}
                aria-label={`${actionButtonStr ?? tryPromptButtonLabel}${
                  tryPromptButtonAriaLabelledby ? ": " : ""
                }`}
                aria-labelledby={tryPromptButtonAriaLabelledby}
                tabIndex={excludeFromTabOrder ? -1 : 0}
              >
                {actionButtonIcon && (
                  <div className={css["action-button-icon"]} slot="icon">
                    {actionButtonIcon}
                  </div>
                )}
                {actionButtonStr ?? tryPromptButtonLabel}
              </Button>
            </div>
          )}
        </Theme>
      </Flex>
    </div>
  );
});
