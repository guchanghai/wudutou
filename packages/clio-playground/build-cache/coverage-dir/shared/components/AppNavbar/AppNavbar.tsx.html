import ChevronLeft from "@spectrum-icons/workflow/ChevronLeft";
import {
  forwardRef,
  type ReactElement,
  type Ref,
  useImperativeHandle,
  useRef,
} from "react";
import { useIntl } from "react-intl";
import { useNavigate } from "react-router-dom";
import { UniversalNavContainer } from "shared/components/UserAvatar/UserAvatar";
import { useAnalyticsContext } from "shared/contexts/AnalyticsContext";
import { useEntitlementContext } from "shared/contexts/EntitlementContext";
import { useAuthContext } from "shared/contexts/IMSContext";
import { useThemeContext } from "shared/contexts/ThemeContext";
import {
  trackSignIn,
  useMarketingScript,
} from "shared/hooks/useMarketingScript";
 
import { AdobeFireflyLogo } from "../AdobeLogo/AdobeLogo";
import { AppTopBannerContainer } from "../AppTopBannerContainer/AppTopBannerContainer.js";
import { EmbedCloseButton } from "../EmbedActionButtons/EmbedCloseButton.js";
import { HeaderWrapper } from "../Header/Header";
import { IconButton } from "../IconButton/IconButton";
import { AppNavBarUtilities } from "./AppNavbarUtilities";
import style from "./css/AppNavbar.module.scss";
 
export interface Props {
  title?: string;
  back?: string | (() => void);
  children?: ReactElement;
  hideBackButton?: boolean;
  hideUpgradeButton?: boolean;
  hideFeedbackButton?: boolean;
  isEmbedContext?: boolean;
  icons?: ReactElement;
  buttons?: ReactElement;
  customNavbarLeft?: ReactElement;
  fireflyLogoClickHandler?: () => void;
}
 
export interface AppNavbarHandle {
  focus(): void;
}
 
export const AppNavbar = forwardRef(
  (
    {
      title,
      back,
      children,
      icons,
      buttons,
      hideBackButton = false,
      hideUpgradeButton = false,
      hideFeedbackButton = false,
      isEmbedContext = false,
      customNavbarLeft,
      fireflyLogoClickHandler,
    }: Props,
    ref: Ref<AppNavbarHandle>,
  ) => {
    const intl = useIntl();
    const { isK12 } = useEntitlementContext();
    const domRef = useRef<HTMLDivElement | null>(null);
    const navigate = useNavigate();
    const { colorTheme } = useThemeContext();
    const { signIn } = useAuthContext();
    useMarketingScript();
 
    const { analytics } = useAnalyticsContext();
 
    function navigateBack() {
      if (!back) {
        navigate("/");
      } else if (typeof back === "string") {
        navigate(back);
      } else {
        back();
      }
    }
 
    /**
     * @returns to-link for app logo. If logo click handler callback is given return empty string, else return "/".
     */
    function getHrefLink() {
      return fireflyLogoClickHandler ? "" : "/";
    }
 
    const handleClickSignIn = async () => {
      trackSignIn("UserAvatarModule");
 
      await analytics.recordClickSignIn();
      signIn();
    };
 
    useImperativeHandle(ref, () => {
      return {
        focus: () => {
          requestAnimationFrame(() => {
            if (document.activeElement !== document.body) {
              return;
            }
            const className = `.${style["navbar-container"]}`;
            const buttonOrLink: HTMLElement | null | undefined =
              domRef.current?.querySelector(
                `${className} [role="button"], ${className} sp-action-button`,
              ) ?? domRef.current?.querySelector(`${className} a[href]`);
            buttonOrLink?.focus();
          });
        },
      };
    });
 
    const NavBarNavigation = () =>
      isEmbedContext || hideBackButton ? null : (
        <div slot="app-menu">
          <IconButton
            icon={<ChevronLeft />}
            onClick={() => navigateBack()}
            quiet
            label={intl.formatMessage({
              id: "@clio/playground:appnavbar-back",
              defaultMessage: "Back",
            })}
            showTooltip
            placement="bottom"
          />
        </div>
      );
 
    const NavBarTitle = () =>
      title ? <span slot="app-menu">{title}</span> : null;
 
    const NavBarLeft = () => {
      if (customNavbarLeft) {
        return customNavbarLeft;
      }
      return (
        <>
          <AdobeFireflyLogo
            iconOnly={true}
            themeColor={colorTheme === "dark" ? "dark" : "light"}
            href={getHrefLink()}
            clickHandler={fireflyLogoClickHandler}
          />
 
          <NavBarNavigation />
 
          <NavBarTitle />
        </>
      );
    };
 
    return (
      <header
        ref={domRef}
        className={style["navbar-container"]}
        data-testid="header-navbar"
      >
        <AppTopBannerContainer />
 
        <HeaderWrapper noDividers={true}>
          <NavBarLeft />
          <AppNavBarUtilities
            icons={icons}
            buttons={buttons}
            children={children}
            hideUpgradeButton={hideUpgradeButton}
            hideDropdown={isK12 || hideFeedbackButton}
          />
 
          <div slot="account-menu">
            <UniversalNavContainer
              onClickSignIn={handleClickSignIn}
              isEditorPage={true}
            />
          </div>
          {isEmbedContext && (
            <div slot="account-menu">
              <EmbedCloseButton />
            </div>
          )}
        </HeaderWrapper>
      </header>
    );
  },
);
