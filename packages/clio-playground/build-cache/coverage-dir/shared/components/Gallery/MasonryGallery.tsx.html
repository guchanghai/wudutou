import Virtualizer, {
  type VirtualizerInputHandles,
} from "@adobe/uxp-optimized/Virtualizer";
import classNames from "classnames";
import { type ReactElement, useEffect, useRef, useState } from "react";
import { createLayoutMasonry } from "shared/components/CustomLayout/layoutMasonry";
import { useIsMobile } from "shared/hooks/useScreenType";
import { useSize } from "shared/hooks/useSize";
import { PageSource } from "shared/types/AnalyticsTypes";
import { type Size } from "shared/types/Geometry.js";
 
import { MarketingBannerWrapper } from "../CustomBanner/MarketingBannerWrapper";
import style from "./css/Gallery.module.scss";
 
type GalleryEntryWithBounds<T> = T & { bounds: DOMRect };
 
export function useGalleryFeedWithLayout<
  T extends { aspectRatio: number; key: string; type?: string },
>(props: {
  items: Array<T>;
  size: Size;
  gap: number;
  margin: number;
  headerHeight?: number;
  footerHeight?: number;
  gridWidth?: GridWidth;
}): Array<GalleryEntryWithBounds<T> | HeaderOrFooter> {
  const isMobile = useIsMobile();
  const {
    items,
    size,
    gap,
    margin,
    headerHeight = 0,
    footerHeight = 0,
    gridWidth,
  } = props;
  const { width } = size;
  const optimumWidth = gridWidth
    ? isMobile
      ? gridWidth.optimumWidth.mobile
      : gridWidth.optimumWidth.browser
    : 250;
  const minWidth = gridWidth
    ? isMobile
      ? gridWidth.minWidth.mobile
      : gridWidth.minWidth.browser
    : 150;
  const layout = createLayoutMasonry({
    optimumWidth: optimumWidth,
    gap,
    minColumns: 1,
    maxColumns: 5,
    minWidth: minWidth,
    getAspect(item: T) {
      return item.aspectRatio;
    },
  });
 
  const bounds = layout({
    width: width - margin * 2,
    height: Number.MAX_SAFE_INTEGER,
    count: items.length,
    items: items,
  });
 
  // Now integrate the rects with bounds with the feed.
  let bottom = 0;
  const entriesWithBounds = items.map((entry, i) => {
    let b: DOMRect = bounds[i];
    b = new DOMRect(b.x + margin, b.y + headerHeight, b.width, b.height);
    bottom = Math.max(bottom, b.y + b.height);
    return { ...entry, bounds: b, type: entry.type ?? "inspiration" };
  });
 
  const fullEntries: Array<GalleryEntryWithBounds<T> | HeaderOrFooter> = [
    ...entriesWithBounds,
  ];
  if (headerHeight) {
    fullEntries.unshift({
      type: "header",
      key: "header",
      bounds: new DOMRect(0, 0, width, headerHeight),
    });
  }
  if (footerHeight && items.length > 0 /* prevent flash when loading */) {
    fullEntries.push({
      type: "footer",
      key: "footer",
      bounds: new DOMRect(0, bottom + gap, width, footerHeight),
    });
  }
  return width > 0 ? fullEntries : [];
}
 
export interface HeaderOrFooter {
  type: "header" | "footer";
  key: "header" | "footer";
  bounds: DOMRect;
}
 
interface GridWidth {
  optimumWidth: {
    mobile: number;
    browser: number;
  };
  minWidth: {
    mobile: number;
    browser: number;
  };
}
 
function isHeaderOrFooter<T extends { type?: string }>(
  item: T | HeaderOrFooter,
): item is HeaderOrFooter {
  return item.type === "header" || item.type === "footer";
}
 
export type GalleryRenderItemProp<T> = T & {
  uid: string;
  excludeFromTabOrder: boolean;
  activeSample: string | undefined;
  setActiveSample: (uid: string | undefined) => void;
};
 
export interface MasonryGalleryProps<T> {
  items: Array<T>;
  gap: number;
  margin: number;
  header?: {
    headerId?: string;
    height: number;
    element?: ReactElement;
    subHeaderId?: string;
  };
  footer?: {
    height: number;
    element?: ReactElement;
  };
  gridWidth?: GridWidth;
  enableScroll?: boolean;
  renderItem?: (props: GalleryRenderItemProp<T>) => ReactElement;
  loadMore?: () => void;
  navigateToEntry?: (item: GalleryEntryWithBounds<T>) => void;
  onResetFocus?: boolean;
  setOnResetFocus?: (onResetFocus: boolean) => void;
  disableUpDown?: boolean;
}
 
export function MasonryGallery<
  T extends { aspectRatio: number; key: string; type?: string },
>(props: MasonryGalleryProps<T>) {
  const {
    gap,
    margin,
    gridWidth,
    navigateToEntry,
    enableScroll = true,
  } = props;
  const containerRef = useRef<HTMLDivElement>(null);
  const virtualizerRef = useRef<VirtualizerInputHandles>(null);
  const size = useSize(containerRef);
  const items = useGalleryFeedWithLayout({
    items: props.items,
    size,
    gap,
    margin,
    headerHeight: props.header?.height ?? margin,
    footerHeight: props.footer?.height ?? margin,
    gridWidth,
  });
 
  const usedKeys: Array<string> = [];
 
  const [focusedGalleryEntryKey, setFocusedGalleryEntryKey] = useState<
    string | null
  >(null);
 
  useEffect(() => {
    if (props.onResetFocus) {
      setActiveSample(undefined);
      props.setOnResetFocus?.(false);
    }
    setFocusedGalleryEntryKey(null);
  }, [props.items]);
 
  function handleFeedFocusCapture(evt: React.FocusEvent<HTMLElement>) {
    const closestArticle = (evt.target as HTMLElement).closest("article");
    if (!closestArticle) {
      return;
    }
    if (closestArticle.dataset.key) {
      setFocusedGalleryEntryKey(closestArticle.dataset.key);
    }
  }
 
  const SCROLL_TO_ITEM_DURATION = 0.04;
  const FOCUS_ADJACENT_ITEM_DELAY = 50;
  let timeoutId: NodeJS.Timeout | null = null;
  function handleFeedKeydown(evt: React.KeyboardEvent<HTMLElement>) {
    const closestArticle = (evt.target as HTMLElement).closest("article");
    if (!closestArticle) {
      return;
    }
    const itemKey = closestArticle.dataset.key;
    const itemIndex = items.findIndex((item) => item.key === itemKey);
    let adjacentItemIndex = -1;
    let adjacentArticle: HTMLElement | null;
    switch (evt.key) {
      case "Enter":
      case " ": // Space
        if (evt.target === closestArticle && navigateToEntry) {
          const item = items.find(
            (item) => item.key === focusedGalleryEntryKey,
          );
          if (item && !isHeaderOrFooter(item)) {
            navigateToEntry(item);
          }
        }
        break;
      case "ArrowDown":
        if (props.disableUpDown) {
          break;
        }
      /* eslint-disable-next-line no-fallthrough --
   enabling fallthrough as if `disableUpDown` is false code in `PageDown` block is valid for current block too. */
      case "ArrowRight":
      case "PageDown":
        evt.preventDefault();
        adjacentItemIndex = itemIndex < items.length - 2 ? itemIndex + 1 : -1;
        break;
      case "ArrowUp":
        if (props.disableUpDown) {
          break;
        }
      /* eslint-disable-next-line no-fallthrough --
   "enabling fallthrough as if `disableUpDown` is false code in `PageUp` block is valid for current block too." */
      case "ArrowLeft":
      case "PageUp":
        evt.preventDefault();
        adjacentItemIndex = itemIndex > 1 ? itemIndex - 1 : -1;
        break;
    }
    if (adjacentItemIndex > 0) {
      const key = items[adjacentItemIndex].key;
      adjacentArticle = containerRef.current!.querySelector(
        `article[data-key="${key}"]`,
      );
      if (adjacentArticle) {
        adjacentArticle.focus();
        setFocusedGalleryEntryKey(key);
        return;
      }
      virtualizerRef.current?.scrollToItem(items[adjacentItemIndex].key, {
        position: 1,
        duration: timeoutId ? 0 : SCROLL_TO_ITEM_DURATION,
      });
      if (timeoutId) {
        clearTimeout(timeoutId);
        timeoutId = null;
      }
      timeoutId = setTimeout(() => {
        adjacentArticle = containerRef.current!.querySelector(
          `article[data-key="${key}"]`,
        )!;
        adjacentArticle.focus();
        setFocusedGalleryEntryKey(key);
      }, FOCUS_ADJACENT_ITEM_DELAY);
    }
  }
 
  // To keep track of currently selected inspire sample (on touch devices)
  const [activeSample, setActiveSample] = useState<string | undefined>(
    undefined,
  );
 
  return (
    <main
      style={{
        height: "100%",
        width: "100%",
        flexDirection: "column",
        display: "flex",
      }}
    >
      <section
        ref={containerRef}
        role="feed"
        onFocusCapture={handleFeedFocusCapture}
        onKeyDownCapture={handleFeedKeydown}
        aria-labelledby={
          props.header?.element && props.header.headerId
            ? props.header.headerId
            : undefined
        }
        aria-describedby={
          props.header?.element && props.header.subHeaderId
            ? props.header.subHeaderId
            : undefined
        }
      >
        <Virtualizer
          data-testid="gallery-feed"
          className={classNames(style.masonryVirtualizer, {
            [style.noScroll]: !enableScroll,
          })}
          ref={virtualizerRef}
          items={items}
          itemKey="key"
          itemType={
            //  item type function controls which items are re-cycled.
            //  items are only recycled within the same type.
            (item: T | HeaderOrFooter) => {
              if (isHeaderOrFooter(item)) {
                return item.key;
              }
              return item.type ?? "inspiration";
            }
          }
          itemRect="bounds"
        >
          {(
            item:
              | (GalleryEntryWithBounds<T> & { title?: string })
              | HeaderOrFooter,
          ) => {
            if (isHeaderOrFooter(item)) {
              if (item.type === "header") {
                usedKeys.length = 0;
                return <div>{props.header?.element}</div>;
              } else {
                return (
                  <div className={style.footer}>{props.footer?.element}</div>
                );
              }
            } else {
              let index = usedKeys.indexOf(item.key);
 
              if (index === -1) {
                index = usedKeys.push(item.key);
              }
 
              const isLastElement =
                item.key === props.items[props.items.length - 1].key;
 
              // the Virtualizer when renders the last element. we should check if more items
              if (isLastElement) {
                if (props.loadMore) {
                  props.loadMore();
                }
              }
              const itemProps = {
                ...item,
                key: item.key,
                uid: item.key,
                excludeFromTabOrder:
                  focusedGalleryEntryKey !== null &&
                  focusedGalleryEntryKey !== item.key,
                activeSample,
                setActiveSample,
              };
              const element = props.renderItem!(itemProps);
              return (
                <article
                  key={item.key}
                  className={style["feed-item"]}
                  aria-label={`${item.title}`}
                  aria-posinset={index}
                  aria-setsize={items.length - 2}
                  tabIndex={
                    focusedGalleryEntryKey === item.key
                      ? 0
                      : focusedGalleryEntryKey === null
                      ? undefined
                      : -1
                  }
                >
                  {element}
                </article>
              );
            }
          }}
        </Virtualizer>
        <MarketingBannerWrapper feedbackId={PageSource.gallery} />
      </section>
    </main>
  );
}
