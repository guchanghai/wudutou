import { action, computed, makeAutoObservable, observable, when } from "mobx";
import { type IStore, StoreLoadingState } from "shared/types/StoreTypes";
 
import { type AuthStore } from "./AuthStore";
 
export class LegalAcknowledgementStore implements IStore {
  @observable _isTouStatusReady = false;
  @observable _showTouModal = false;
  @observable _touAccepted = false;
  @observable _touCanceled = false;
  // eslint-disable-next-line @typescript-eslint/no-empty-function -- no-op fn
  @observable callback: () => void = () => {};
  @observable pendingPromise: Promise<void> | null = null;
  @observable resolvePromise: () => void =
    // eslint-disable-next-line @typescript-eslint/no-empty-function -- no-op func
    () => {};
  @observable private _loadingState: StoreLoadingState =
    StoreLoadingState.Uninitialized;
 
  constructor(private authStore: AuthStore) {
    makeAutoObservable(this);
    when(
      () => this.isTouStatusReady,
      () => this.resolvePromise(),
    );
  }
 
  @action
  private setLoadingState = (loadingState: StoreLoadingState) => {
    this._loadingState = loadingState;
  };
 
  @computed
  get loadingState(): StoreLoadingState {
    return this._loadingState;
  }
 
  @computed
  get isLoaded(): boolean {
    return this._loadingState === StoreLoadingState.Loaded;
  }
 
  setIsTouStatusReady = (isReady: boolean) => {
    this._isTouStatusReady = isReady;
  };
 
  @action
  initialize = async () => {
    this.setLoadingState(StoreLoadingState.Loading);
    this.setLoadingState(StoreLoadingState.Loaded);
    return await Promise.resolve();
  };
 
  @action
  setShowTouModal = (show: boolean) => {
    this._showTouModal = show;
  };
 
  @action
  setCallback = (callback: () => void) => {
    this.callback = callback;
  };
 
  @action
  setTouAccepted = (accepted: boolean) => {
    this._touAccepted = accepted;
    this.resolvePromise();
  };
 
  @action
  setTouCanceled = (canceled: boolean) => {
    this._touCanceled = canceled;
    this.resolvePromise();
  };
 
  @computed get isTouStatusReady() {
    return this._isTouStatusReady;
  }
 
  @computed get showTouModal() {
    return this._showTouModal;
  }
 
  @computed get touAccepted() {
    return this._touAccepted;
  }
 
  @computed get touCanceled() {
    return this._touCanceled;
  }
 
  /**
   * Checks if the Terms of Use (TOU) have been accepted.
   *
   * If the user is not signed in, initiates the sign-in process with any state that needs to be persisted for page.
   * If the user is signed in and the TOU status is ready, resolves the pending promise.
   * If the TOU have not been accepted, shows the TOU modal.
   *
   * @param pagePersistedState - An optional state object to be passed to the sign-in method.
   * @returns A promise that resolves when the TOU status is ready.
   */
  @action
  checkTOUAccepted = (pagePersistedState?: Record<string, unknown>) => {
    if (this.pendingPromise === null) {
      this.pendingPromise = new Promise((resolve) => {
        this.resolvePromise = resolve;
      });
    }
    if (!this.authStore.isSignedIn) {
      this.authStore.signIn(undefined, pagePersistedState);
    } else {
      if (this.isTouStatusReady) {
        if (this.touAccepted) {
          this.resolvePromise();
        } else {
          this.resolvePromise();
          this.setShowTouModal(true);
        }
      }
    }
    return this.pendingPromise;
  };
 
  @computed shouldShowTouModal = (
    authorizedUser: boolean,
    loadCompleted: boolean,
  ) => {
    return (
      this.isTouStatusReady &&
      ((authorizedUser && !loadCompleted) || // for authorizedUser but blocked by entitlement, we request to show TOU modal
        (this.showTouModal && !this.touAccepted) ||
        !this.touAccepted)
    );
  };
 
  @computed shouldShowTouBanner = (authorizedUser: boolean) => {
    if (this.isTouStatusReady && authorizedUser) {
      return this.touCanceled;
    }
  };
}
