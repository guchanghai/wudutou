import { action, makeAutoObservable, observable, reaction } from "mobx";
import { deployEnvironment } from "shared/config/BuildMode";
import { makeLoggable } from "shared/data/MobxFunctions";
import {
  type FireflyDevEndpointTypes,
  FireflyEndpoints,
  type FireflyEndpointTypes,
  getFireflyEndpointType,
} from "shared/services/config/FireflyEndpoints";
import { Text2ImageServiceAdapterV2SDK } from "shared/services/firefly/clio2/Text2ImageServiceAdapterV2SDK";
import { type Text2ImageAPI } from "shared/services/firefly/FireflyText2ImageAPI";
import { type FloodgateFlagsMap } from "shared/services/floodgate/FloodgateTypes.js";
import { type Text2ImageEditModel } from "shared/types/Text2ImageModelDefaults";
import { type Text2ImageResults } from "shared/types/Text2ImageTypes";
import {
  type ClioError,
  type ClioErrorCode,
  getClioError,
} from "shared/utils/ClioError";
 
import { type AuthStore } from "./AuthStore.js";
import { type PersonalizationStore } from "./PersonalizationStore.js";
 
export interface IFireflyServiceFeatureFlags {
  enableHardCap: boolean;
  enableColligoV2: boolean;
  enableColligoV3: boolean;
  enableParametricComparison: boolean;
  isAltEnvironment: boolean;
}
 
export type EndpointRecord<T> = Record<FireflyEndpointTypes, T> &
  Partial<Record<FireflyDevEndpointTypes, T>>;
 
/**
 * FireflyService is responsible for generate images from Firefly service.
 */
export class FireflyService {
  private _text2ImageAdapters: EndpointRecord<Text2ImageAPI | undefined>;
  private _featureFlags?: IFireflyServiceFeatureFlags;
 
  @observable.ref fireflyServiceErrors: Array<ClioErrorCode> = [];
 
  constructor(
    private _authStore: AuthStore,
    private _personalizationStore: PersonalizationStore,
  ) {
    makeAutoObservable(this);
    makeLoggable(this);
    this.initialize();
    this._text2ImageAdapters = {
      "t2i-v1": undefined,
      "clio-v2": undefined,
      "clio-v2-structure-match": undefined,
      "clio-v3": undefined,
      "clio-v2-cme": undefined,
      "clio-v3-cme": undefined,
    };
    this._updateFeatureFlags(
      this._personalizationStore.floodgateFlagsWithOverrides,
    );
  }
  @action
  initialize = () => {
    reaction(
      () => [this._authStore.isLoaded, this._authStore.accessToken],
      () => {
        if (!this._authStore.isLoaded) {
          return;
        }
        this._updateAccessToken(this._authStore.accessToken);
      },
    );
 
    reaction(
      () => [this._personalizationStore.floodgateFlagsWithOverrides],
      () => {
        if (!this._personalizationStore.isFloodgateLoaded) {
          return;
        }
        this._updateFeatureFlags(
          this._personalizationStore.floodgateFlagsWithOverrides,
        );
      },
    );
  };
 
  @action
  _setServiceError = (error: ClioError) => {
    this.fireflyServiceErrors = [...this.fireflyServiceErrors, error.code];
  };
 
  @action
  resetServiceErrors = () => {
    if (this.fireflyServiceErrors.length > 0) {
      this.fireflyServiceErrors = [];
    }
  };
 
  private _getText2ImageAdapter(
    endpointType: FireflyEndpointTypes | FireflyDevEndpointTypes,
  ): Text2ImageAPI {
    let adapter = this._text2ImageAdapters[endpointType];
    if (!adapter) {
      const accessToken = this._authStore.accessToken;
      const clientId = this._authStore.config.clientId;
      const endpoint = FireflyEndpoints[endpointType];
      if (endpoint && endpointType !== "t2i-v1") {
        const env = deployEnvironment();
        const isAltEnvironment = this._featureFlags?.isAltEnvironment ?? false;
        const generateUrl = isAltEnvironment
          ? endpoint.generate.alternative?.[env]
          : endpoint.generate[env];
        const baseUrl = generateUrl ?? endpoint.generate[env];
        const uploadUrl = endpoint.upload[env];
 
        if (!baseUrl || !uploadUrl) {
          throw new Error(
            `Unable to use '${endpointType}' in '${env}' environment.`,
          );
        }
 
        const fireflyApiClientConfig = {
          baseUrl,
          accessToken,
          clientId,
          urlOverrides: {
            "/v2/storage/image": uploadUrl,
          },
        };
        // use V2 API to call both V2 and V3 endpoint
        adapter = new Text2ImageServiceAdapterV2SDK(fireflyApiClientConfig);
      } else {
        console.error("SPL(V1) is not supported");
        throw new Error("SPL(V1) is not supported");
      }
      this._text2ImageAdapters[endpointType] = adapter;
    }
 
    return adapter;
  }
 
  private _updateAccessToken(accessToken?: string) {
    if (!accessToken) {
      return;
    }
    for (const type in this._text2ImageAdapters) {
      const endpointType = type as FireflyEndpointTypes;
      const adapter = this._text2ImageAdapters[endpointType];
      if (adapter) {
        adapter.updateAuthToken(accessToken);
      }
    }
  }
 
  private _updateFeatureFlags(featureFlags?: FloodgateFlagsMap) {
    const enableHardCap = Boolean(
      featureFlags?.["firefly-commercialization-hard-cap"],
    );
    const enableColligoV2 = Boolean(featureFlags?.["firefly-apis-colligo-v2"]);
    const enableColligoV3 = Boolean(featureFlags?.["firefly-apis-colligo-v3"]);
    const enableParametricComparison = Boolean(
      featureFlags?.["clio-parametric-comparison"],
    );
    const isAltEnvironment = Boolean(featureFlags?.["firefly-alt-environment"]);
    this._featureFlags = {
      enableHardCap,
      enableColligoV2,
      enableColligoV3,
      enableParametricComparison,
      isAltEnvironment,
    };
  }
 
  public async text2ImageForEditModel(
    model: Text2ImageEditModel,
    signal: AbortSignal,
  ): Promise<Text2ImageResults> {
    const enableHardCap = this._featureFlags?.enableHardCap ?? false;
    const enableColligoV2 = this._featureFlags?.enableColligoV2 ?? false;
    const enableColligoV3 = this._featureFlags?.enableColligoV3 ?? false;
    const enableParametricComparison =
      this._featureFlags?.enableParametricComparison ?? false;
    try {
      const endpointType = getFireflyEndpointType(
        model,
        enableColligoV2,
        enableColligoV3,
        enableParametricComparison,
      );
      const adapter = this._getText2ImageAdapter(endpointType);
      if (!adapter.text2ImageForEditModel) {
        throw new Error("Endpoint does not support text2ImageForEditModel");
      }
      const result = await adapter.text2ImageForEditModel(model, signal);
 
      if (!enableHardCap) {
        result.consumptionStatus = undefined;
        result.appliedUntilDate = undefined;
      }
 
      return result;
    } catch (e) {
      this._handleServiceError(e);
      throw e;
    }
  }
  private _handleServiceError(e: unknown) {
    const clioError: ClioError = getClioError(e as Error);
 
    // Verify if the error is due invalid token
    if (clioError.code === "INVALID_TOKEN") {
      this._authStore.signIn();
    }
 
    this._setServiceError(clioError);
  }
}
