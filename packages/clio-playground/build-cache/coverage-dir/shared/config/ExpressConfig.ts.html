export const getExpressGenerativeImageUrl = (
  params: ExpressGenerativeImageParams,
) => {
  const {
    prompt,
    seed,
    styles,
    aspectRatio,
    fireflyImg,
    height,
    width,
    intent,
    editAction,
    expressFeature,
    useMobileDeeplink,
  } = params;
 
  const newProjectUrl = useMobileDeeplink
    ? getExpressNewProjectUrlMobile()
    : getExpressNewProjectUrl();
  const promoId = getPromoId(intent, expressFeature, editAction);
  const entryPoint = getEntryPoint(expressFeature);
 
  return `${newProjectUrl}?prompt=${encodeURIComponent(
    prompt,
  )}&seed=${seed}&firefly-styles=${encodeURIComponent(
    styles,
  )}&aspectRatio=${aspectRatio}&referrer=firefly&firefly-img=${fireflyImg}&height=${height}&width=${width}&intent=${intent}${
    editAction ? `&edit-action=${ExpressEditActionLabels[editAction]}` : ""
  }&entryPoint=${entryPoint}&mv=other&mv2=ccx${
    promoId ? `&promoid=${promoId}` : ""
  }`;
};
 
export const getExpressBitmapUrl = (params: ExpressBitmapParams) => {
  const {
    img,
    height,
    width,
    expressFeature,
    intent,
    editAction,
    useMobileDeeplink,
  } = params;
 
  const newProjectUrl = useMobileDeeplink
    ? getExpressNewProjectUrlMobile()
    : getExpressNewProjectUrl();
  const promoId = getPromoId(intent, expressFeature, editAction);
  const neuralFeature = getNeuralFeature(expressFeature);
  const entryPoint = getEntryPoint(expressFeature);
 
  return `${newProjectUrl}?firefly-img=${img}&referrer=firefly&height=${height}&width=${width}&intent=${intent}${
    editAction ? `&edit-action=${ExpressEditActionLabels[editAction]}` : ""
  }&entryPoint=${entryPoint}&neural-feature=${neuralFeature}&mv=other&mv2=ccx${
    promoId ? `&promoid=${promoId}` : ""
  }`;
};
 
function getNeuralFeature(feature: ExpressFeature): string {
  return ExpressNeuralFeature[feature];
}
 
function getEntryPoint(feature: ExpressFeature): string {
  return ExpressEntryPoint[feature];
}
 
function getPromoId(
  intent: ExpressIntent,
  feature: ExpressFeature,
  editAction?: ExpressEditAction,
): string {
  let promoId = promoIds[intent][feature];
  if (editAction) {
    promoId = promoIds[editAction][feature];
  }
 
  return promoId ?? "";
}
